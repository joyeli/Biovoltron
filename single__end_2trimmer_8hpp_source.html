79 auto detect_adapter(const Tailor<>& tailor,
80 const std::vector<R>& reads,
81 const bool sensitive) {
82
83 /* collect tails by calling Tailor */
84 /* TODO use SeqView here, but we should change to SeqType */
85 std::vector<SeqType> tails;
86 for (int i = 0; i < (int) reads.size() && (int) tails.size() < 3000; i++) {
87 auto align_res = tailor.search(reads[i]);
88 if (align_res.tail_pos >= 0) {
89 auto s = align_res.seq.substr(align_res.tail_pos);
90 if (!s.empty()) {
91 tails.emplace_back(s);
92 }
93 }
94 }
95 std::vector<SeqView> tail_views;
96 std::ranges::transform(tails, std::back_inserter(tail_views), [](auto &s) {
97 return SeqView(s);
98 });
99 auto adapter = assembler.assemble(tail_views, sensitive);
100 SPDLOG_DEBUG("tail size = {}", tails.size());
101
102 if (adapter.empty()) {
103 spdlog::info("Unable to detect adapter, use default adapter");
104 adapter = param.DEFAULT_ADAPTER1;
105 }
106 spdlog::info("Detected adpater = {}", adapter);
107 return adapter;
108 }

110 auto call_skewer(const std::filesystem::path &reads_path,
111 const SeqView adapter,
112 const std::filesystem::path &output_path,
113 const std::size_t thread_num,
114 const bool sensitive) {
115
116 spdlog::info("Call skewer for adapter trimming");
117
118 auto skewer_out = std::filesystem::temp_directory_path() / "skewer_output";
119 std::vector<std::string> skewer_argv;
120 skewer_argv.emplace_back("skewer"); // main program
121 skewer_argv.emplace_back(reads_path.string());
122 skewer_argv.emplace_back("-o");
123 skewer_argv.emplace_back(skewer_out.string());
124 skewer_argv.emplace_back("-l"); /* minimum adapter length */
125 skewer_argv.emplace_back(std::to_string(param.MIN_ADAPTER_LEN));
126 skewer_argv.emplace_back("-t"); /* thread number */
127 skewer_argv.emplace_back(std::to_string(thread_num));
128
129 if (sensitive) {
130 skewer_argv.emplace_back("-r"); /* TODO: survey what is this */
131 skewer_argv.emplace_back("0.2");
132 }
133 if (adapter.size()) {
134 skewer_argv.emplace_back("-x");
135 if constexpr (R::encoded) {
136 skewer_argv.emplace_back(Codec::to_string(adapter));
137 } else {
138 skewer_argv.emplace_back(adapter);
139 }
140 }
141
142 auto skewer_argc = skewer_argv.size();
143 SPDLOG_DEBUG("Called skewer: argc = {}", skewer_argc);
144 {
145 std::string cmd;
146 std::ranges::for_each(skewer_argv, [&](auto& s) { cmd += s + ' '; });
147 SPDLOG_DEBUG("skewer cmd = {}", cmd);
148 }
149 std::vector<const char*> real_argv(skewer_argc);
150 std::ranges::for_each(std::views::iota(0u, skewer_argc), [&](auto idx) {
151 real_argv[idx] = skewer_argv[idx].data();
152 });
153 skewer::main(skewer_argc, real_argv.data());
154 skewer_out += "-trimmed.fastq";
155 assert(std::filesystem::exists(skewer_out));
156 std::filesystem::copy(skewer_out, output_path,
157 std::filesystem::copy_options::overwrite_existing);
158 std::filesystem::remove(skewer_out);
159 }

266 auto input_format = detail::parse_file_format(reads_path);
