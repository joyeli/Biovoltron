79  auto detect_adapter(const Tailor<>& tailor,
80  const std::vector<R>& reads,
81  const bool sensitive) {
82 
83  /* collect tails by calling Tailor */
84  /* TODO use SeqView here, but we should change to SeqType */
85  std::vector<SeqType> tails;
86  for (int i = 0; i < (int) reads.size() && (int) tails.size() < 3000; i++) {
87  auto align_res = tailor.search(reads[i]);
88  if (align_res.tail_pos >= 0) {
89  auto s = align_res.seq.substr(align_res.tail_pos);
90  if (!s.empty()) {
91  tails.emplace_back(s);
92  }
93  }
94  }
95  std::vector<SeqView> tail_views;
96  std::ranges::transform(tails, std::back_inserter(tail_views), [](auto &s) {
97  return SeqView(s);
98  });
99  auto adapter = assembler.assemble(tail_views, sensitive);
100  SPDLOG_DEBUG("tail size = {}", tails.size());
101 
102  if (adapter.empty()) {
103  spdlog::info("Unable to detect adapter, use default adapter");
104  adapter = param.DEFAULT_ADAPTER1;
105  }
106  spdlog::info("Detected adpater = {}", adapter);
107  return adapter;
