159  auto find_possible_adapter_pos(SeqView seq,
160  SeqView adapter) const noexcept
161  -> std::vector<std::size_t> {
162  auto possible_pos = std::vector<std::size_t> {};
163 
164  const auto read_size = std::min(detail::BASE_IN_VECTOR, seq.size());
165  const auto cmp_len = std::min(param.RC_CHECK_LEN, read_size);
166  const auto subseq = seq.substr(0, read_size);
167  auto seq_vec = detail::make_simd_vector(subseq, 1)[0];
168  const auto adapter_vec = detail::make_simd_vector(adapter, 1)[0];
169 
170  /* interate all possible subsequence */
171  for (auto i = 0U; i <= read_size - cmp_len; i++) {
172  auto simd_sim = detail::cal_similarity(seq_vec, adapter_vec, cmp_len);
173  if (simd_sim >= param.TAIL_MATCH_RATIO) {
174  possible_pos.emplace_back(i + cmp_len);
175  }
176  seq_vec = detail::shift_left(seq_vec);
177  }
184  if (seq.size() > detail::BASE_IN_VECTOR) {
185  auto base_pos = detail::BASE_IN_VECTOR - param.RC_CHECK_LEN + 1;
186  auto&& r = find_possible_adapter_pos(seq.substr(base_pos), adapter);
187  std::ranges::transform(r, std::back_inserter(possible_pos),
188  [&](const auto& idx) {
189  return idx + base_pos;
190  });
191  }
192  return possible_pos;
193  }

209  auto get_adapter_pos(const SeqView forward_seq, const SeqView reverse_seq,
210  const std::vector<std::size_t>& intersect_pos)
211  const noexcept
212  -> std::size_t {
213 
214  auto forward_rc = Codec::rev_comp(forward_seq);
215 
216  /* we only need to make simd vector of reverse read for once */
217  auto rev_vec = detail::make_simd_vector(reverse_seq);
218 
219  for (auto fwd_rc_view = SeqView(forward_rc);
220  auto pos : intersect_pos | std::views::reverse) {
221 
222  auto fwd_rc_subseq = fwd_rc_view.substr(fwd_rc_view.size() - pos);
223  auto fwd_rc_subseq_vec = detail::make_simd_vector(fwd_rc_subseq);
224  auto match = 0;
225  for (auto i = 0u; i < fwd_rc_subseq_vec.size(); i++) {
226  auto cmp_len = std::min(
227  detail::BASE_IN_VECTOR,
228  pos - i * detail::BASE_IN_VECTOR
229  );
230  match += detail::cal_match(fwd_rc_subseq_vec[i], rev_vec[i], cmp_len);
231  }
232  if (static_cast<double>(match) / pos >= param.DNA_MATCH_RATIO) {
233  return pos;
234  }
235  }
236  return static_cast<std::size_t>(-1);
237  }

261  auto get_trim_pos(const SeqView forward_seq, const SeqView reverse_seq,
262  const std::vector<std::size_t>& intersect_pos,
263  const SeqView forward_adapter,
264  const SeqView reverse_adapter)
265  const noexcept -> std::size_t {
266 
267  auto forward_rc = Codec::rev_comp(forward_seq);
268 
269  /* we only need to make simd vector of reverse read for once */
270  auto rev_vec = detail::make_simd_vector(reverse_seq);
271  auto fwd_adapter_vec = detail::make_simd_vector(forward_adapter, 1)[0];
272  auto rev_adapter_vec = detail::make_simd_vector(reverse_adapter, 1)[0];
273 
274  /* like `get_adapter_pos()`, we perform SS check here */
275  for (auto fwd_rc_view = SeqView(forward_rc);
276  auto pos : intersect_pos | std::views::reverse) {
277  if (pos == forward_seq.size()) {
278  continue;
279  }
280  auto fwd_rc_subseq = fwd_rc_view.substr(forward_rc.size() - pos);
281  auto fwd_rc_subseq_vec = detail::make_simd_vector(fwd_rc_subseq);
282 
283  auto match = 0;
284  for (auto i = 0u; i < fwd_rc_subseq_vec.size(); i++) {
285  auto cmp_len = std::min(
286  detail::BASE_IN_VECTOR,
287  pos - i * detail::BASE_IN_VECTOR
288  );
289  match += detail::cal_match(fwd_rc_subseq_vec[i], rev_vec[i], cmp_len);
290  }
291 
292  /* pass the SS check, perform AS check */
293  if (static_cast<double>(match) / pos >= param.DNA_MATCH_RATIO) {
294  auto fwd_subseq = forward_seq.substr(pos);
295  auto rev_subseq = reverse_seq.substr(pos);
296 
297  auto fwd_subseq_vec = detail::make_simd_vector(fwd_subseq, 1)[0];
298  auto rev_subseq_vec = detail::make_simd_vector(rev_subseq, 1)[0];
299  auto len = std::min(fwd_subseq.size(), forward_adapter.size());
300 
306  if (detail::cal_similarity(fwd_subseq_vec, fwd_adapter_vec, len)
307  >= param.ADAPTER_MATCH_RATIO ||
308  detail::cal_similarity(rev_subseq_vec, rev_adapter_vec, len)
309  >= param.ADAPTER_MATCH_RATIO) {
310  return pos;
311  }
312  }
313  }
314 
319  if (intersect_pos.empty()) {
320  auto find_head_adapter_pos = [&](const SeqView seq,
321  const detail::SIMD_VECTOR& adapter_vec,
322  const std::size_t adapter_size) {
323  auto seq_vec = detail::make_simd_vector(seq, 1)[0];
324  auto cmp_len = std::min(param.HEAD_ADAPTER_CHECK_LEN, adapter_size);
325  for (auto pos : std::views::iota(0UL, param.HEAD_ADAPTER_CHECK_LEN)) {
326  cmp_len = std::min(seq.size() - pos, cmp_len);
327  if (cmp_len < param.MIN_ADAPTER_LEN) {
328  break;
329  }
330  auto similarity = detail::cal_similarity(
331  seq_vec, adapter_vec, cmp_len
332  );
333  if (similarity >= param.ADAPTER_MATCH_RATIO) {
334  return pos;
335  }
336  seq_vec = detail::shift_left(seq_vec);
337  }
338  return static_cast<std::size_t>(-1);
339  };
340 
341  // ? figure out the origin implementation look like this
342  // from 32 to min_adapter_len, check the similarity of both sequence,
343  // check that whether adapter is at the head, then trim it.
344  {
345  auto forward_subseq = forward_seq.substr(
346  0, 2 * param.HEAD_ADAPTER_CHECK_LEN);
347  auto pos = find_head_adapter_pos(forward_subseq, fwd_adapter_vec,
348  forward_adapter.size());
349  if (pos != static_cast<std::size_t>(-1)) {
350  return pos;
351  }
352  }
353  {
354  auto reverse_subseq = reverse_seq.substr(
355  0, 2 * param.HEAD_ADAPTER_CHECK_LEN);
356  auto pos = find_head_adapter_pos(reverse_subseq, rev_adapter_vec,
357  reverse_adapter.size());
358  if (pos != static_cast<std::size_t>(-1)) {
359  return pos;
360  }
361  }
362  }
363 
364  return forward_seq.size();
365  }

411  auto get_possible_intersect(SeqView fwd_seq,
412  SeqView rev_seq) const noexcept
413  -> std::vector<std::size_t> {
414  auto seq_pair = std::array { fwd_seq, rev_seq };
415  auto possible_pos = std::array<std::vector<std::size_t>, 2>{};
416 
417  for (auto s = 0; s <= 1; s++) {
418  auto rc = Codec::rev_comp(seq_pair[s].substr(0, param.RC_CHECK_LEN));
419  possible_pos[s] = find_possible_adapter_pos(seq_pair[s ^ 1], rc);
420  }
421 
422  auto intersect_pos = std::vector<std::size_t> {};
423  std::ranges::set_intersection(possible_pos[0],
424  possible_pos[1],
425  std::back_inserter(intersect_pos));
426  return intersect_pos;
427  };

437  auto detect_adapter(const std::vector<R>& forward_reads,
438  const std::vector<R>& reverse_reads,
439  bool sensitive) noexcept
440  -> std::pair<SeqType, SeqType> {
441 
442  auto forward_tails = std::vector<SeqView> {};
443  auto reverse_tails = std::vector<SeqView> {};
444  auto sz = forward_reads.size();
445 
446  spdlog::info("Collect tails from forward reads and reverse reads...");
447  for (auto i = 0u; i < sz; i++) {
448  auto fwd_seq = SeqView(forward_reads[i].seq);
449  auto rev_seq = SeqView(reverse_reads[i].seq);
450  auto intersect_pos = get_possible_intersect(fwd_seq, rev_seq);
451 
452  /* No intersect positions, skip this reads */
453  if (intersect_pos.size() == 0) {
454  continue;
455  }
456 
457  auto pos = get_adapter_pos(fwd_seq, rev_seq, intersect_pos);
458 
459  /* cannot find proper adapter position */
460  if (pos == static_cast<std::size_t>(-1) || pos == fwd_seq.size()) {
461  continue;
462  }
463 
464  forward_tails.push_back(fwd_seq.substr(pos));
465  reverse_tails.push_back(rev_seq.substr(pos));
466  }
467  spdlog::info("Done. Start detect adapter from collected tails");
468  SPDLOG_DEBUG("forward tails size = {}", forward_tails.size());
469  SPDLOG_DEBUG("reverse tails size = {}", reverse_tails.size());
470 
471  spdlog::info("Detect forward adapter...");
472  auto forward_adapter = assembler.assemble(forward_tails, sensitive);
473  spdlog::info("Detect reverse adapter...");
474  auto reverse_adapter = assembler.assemble(reverse_tails, sensitive);
475 
476  /* unable to assemble adapter, use default adapter */
477  if (forward_adapter.empty() || reverse_adapter.empty()) {
478  spdlog::info("Unable to detect adapter in reads");
479  spdlog::info("Use default adapter");
480  if constexpr (R::encoded) {
481  forward_adapter = Codec::to_istring(param.DEFAULT_ADAPTER1);
482  reverse_adapter = Codec::to_istring(param.DEFAULT_ADAPTER2);
483  } else {
484  forward_adapter = param.DEFAULT_ADAPTER1;
485  reverse_adapter = param.DEFAULT_ADAPTER2;
486  }
487  }
488 
489  /* trim the adapter to same length */
490  if (forward_adapter.size() > reverse_adapter.size()) {
491  forward_adapter.resize(reverse_adapter.size());
492  } else if (forward_adapter.size() < reverse_adapter.size()) {
493  reverse_adapter.resize(forward_adapter.size());
494  }
495 
496  spdlog::info("The forward adapter = {}", forward_adapter);
497  spdlog::info("The reverse adapter = {}", reverse_adapter);
498 
499  return std::make_pair(forward_adapter, reverse_adapter);
500  }

512  auto trim_read(R& forward_read, R& reverse_read,
513  const SeqType& forward_adapter, const SeqType& reverse_adapter)
514  const noexcept
515  -> void {
516  auto fwd_seq_view = SeqView(forward_read.seq);

518 
519  auto intersect_pos = get_possible_intersect(fwd_seq_view, rev_seq_view);
520

522  intersect_pos,
523  forward_adapter, reverse_adapter);
524 
525  if (trim_pos < param.MIN_READ_LEN) {
526  trim_pos = 0;
527  }
528  forward_read.seq.resize(trim_pos);
529  reverse_read.seq.resize(trim_pos);
530  if constexpr (std::same_as<R, FastqRecord<R::encoded>>) {
531  forward_read.qual.resize(trim_pos);
532  reverse_read.qual.resize(trim_pos);
533  }
534  }

548  template<class IStream, class OStream>
549  auto asio_trim(IStream& forward_fin, IStream& reverse_fin,
550  OStream& forward_fout, OStream& reverse_fout,
551  int thread_num, const bool sensitive) {
552 
553  using BufType = std::vector<R>;
554 
555  auto tmp_dir = std::filesystem::temp_directory_path();
556  auto fwd_tmp_dir = tmp_dir / "fwd";
557  auto rev_tmp_dir = tmp_dir / "rev";
558  std::filesystem::remove_all(fwd_tmp_dir);
559  std::filesystem::remove_all(rev_tmp_dir);
560  if (!std::filesystem::create_directory(fwd_tmp_dir) ||
561  !std::filesystem::create_directory(rev_tmp_dir)) {
562  SPDLOG_ERROR("Cannot create temporary directory {} and {}", fwd_tmp_dir.string(), rev_tmp_dir.string());
563  throw std::runtime_error("Cannot create temporary directory");
564  }
565  auto ios = boost::asio::io_context {};
566  auto guard = boost::asio::make_work_guard(ios);
567  auto total_batch = -1LL;
568  auto writed_batch = std::set<std::size_t> {};
569  auto writed_batch_mtx = std::mutex {};
570 
571 
572  std::vector<std::jthread> threads(thread_num - 1);
573  for (auto i = 0; i < thread_num - 1; i++) {
574  threads.emplace_back(std::jthread([&ios]() {
575  ios.run();
576  }));
577  }
578 
579  SPDLOG_DEBUG("Start Trimming by using Asychronous IO");
580 
581  SamHeader fwd_header, rev_header;
582  if constexpr (std::same_as<IStream, IBamStream>) {
583  forward_fin >> fwd_header;
584  reverse_fin >> rev_header;
585  }
586  auto write_all_files = [&]() {
587  auto batch = 0LL;
588  auto final_write = [](std::filesystem::path p, OStream& fout) {
589  auto fin = IStream(p);
590  assert(fin.is_open());
591  if constexpr (std::same_as<OStream, OBamStream>) {
592  SamRecord<false> r;
593  while (fin >> r) {
594  fout << r;
595  }
596  } else {
597  fout << fin.rdbuf() << std::flush;
598  }
599  fin.close();
600  std::filesystem::remove(p);
601  };
602  if constexpr (std::same_as<OStream, OBamStream>) {
603  forward_fout << fwd_header;
604  reverse_fout << rev_header;
605  }
606  while (1) {
607  while (writed_batch.contains(batch)) {
608  auto fwd_path = fwd_tmp_dir / std::to_string(batch);
609  auto rev_path = rev_tmp_dir / std::to_string(batch);
610  // SPDLOG_DEBUG("[write all] write {}, {}", fwd_path, rev_path);
611  final_write(fwd_path, forward_fout);
612  final_write(rev_path, reverse_fout);
613  batch++;
614  }
615  if (static_cast<int>(writed_batch.size()) == total_batch) {
616  assert(total_batch == batch);
617  // SPDLOG_DEBUG("[write all] Write files finished, batch = {}", batch);
618  break;
619  }
620  std::this_thread::sleep_for(1ms);
621  }
622  };

624  auto asio_read = [&](const std::size_t read_count) {
625  auto fwd_buf = BufType();
626  auto rev_buf = BufType();
627  fwd_buf.reserve(read_count);
628  rev_buf.reserve(read_count);
629  bool fwd_eof = false, rev_eof = false;
630  for (auto i = 0u; i < read_count; i++) {
631  if (R forward_read; !(forward_fin >> forward_read)) {
632  fwd_eof = true;
633  } else {
634  fwd_buf.emplace_back(std::move(forward_read));
635  }
636  if (R reverse_read; !(reverse_fin >> reverse_read)) {
637  rev_eof = true;
638  } else {
639  rev_buf.emplace_back(std::move(reverse_read));
640  }
641  if (fwd_eof != rev_eof) {
642  SPDLOG_ERROR("The forward reads file and reverse reads files doesn't contain same amount reads");
643  throw std::runtime_error("The forward reads file and reverse reads files doesn't contain same amount reads");
644  } else if (fwd_eof) {
645  return std::make_tuple(true, std::move(fwd_buf), std::move(rev_buf));
646  }
647  }
648  return std::make_tuple(false, std::move(fwd_buf), std::move(rev_buf));
649  };

651  auto asio_write = [&](const std::size_t batch,
652  const BufType& fwd_buf, const BufType& rev_buf) {
653  auto write_to_tmp = [](std::filesystem::path p, const BufType& buf) {
654  auto fout = std::ofstream(p, std::ios::binary);
655  if (!fout.is_open()) {
656  SPDLOG_ERROR("Cannot open tmp file {}", p.string());
657  throw std::runtime_error("Cannot open tmp file");
658  }
659  for (const auto& r : buf) {
660  fout << r << '\n';
661  }
662  };
663 
664  auto fwd_path = fwd_tmp_dir / std::to_string(batch);
665  auto rev_path = rev_tmp_dir / std::to_string(batch);
666  write_to_tmp(fwd_path, fwd_buf);
667  write_to_tmp(rev_path, rev_buf);
668  auto lock = std::scoped_lock<std::mutex>(writed_batch_mtx);
669  writed_batch.insert(batch);
670  };

672  auto&& [_, fwd_reads, rev_reads] = asio_read(param.DETECT_READS);
673  auto [forward_adapter, reverse_adapter]
674  = detect_adapter(fwd_reads, rev_reads, sensitive);

676  auto trim_reads = [&](const std::size_t batch,
677  BufType& fwd_buf, BufType& rev_buf) {
678  auto sz = fwd_buf.size();
679  for (auto i = 0u; i < sz; i++) {
680  trim_read(fwd_buf[i], rev_buf[i], forward_adapter, reverse_adapter);
681  }
682  asio_write(batch, fwd_buf, rev_buf);
683  };

685  auto pipeline = [&]() {
686  auto batch = 0;
687  auto bg = std::chrono::high_resolution_clock::now();
688  while (true) {
689  auto&& [eof, fwd_buf, rev_buf] = asio_read(param.ASIO_BUFFER_SIZE);
690  if (fwd_buf.empty()) {
691  break;
692  }
693  // SPDLOG_DEBUG("[Pipeline] Read batch {} finished", batch);
694  boost::asio::post(ios, std::bind(trim_reads, batch,
695  std::move(fwd_buf),
696  std::move(rev_buf)));
697  batch++;
698  if (eof) {
699  forward_fin.close();
700  reverse_fin.close();
701  break;
702  }
703  }
704  auto ed = std::chrono::high_resolution_clock::now();
705  SPDLOG_DEBUG("[Asio trim] read files takes {} ms", (ed - bg) / 1ms);
706  total_batch = batch;
707 
708  boost::asio::post(ios, [&guard]() { guard.reset(); });
709  };
710 
711  if constexpr (std::same_as<IStream, IBamStream>) {
712  forward_fin.to_begin();
713  reverse_fin.to_begin();
714  } else {
715  forward_fin.seekg(0);
716  reverse_fin.seekg(0);
717  }
718 
719  spdlog::info("Start trimming...");
720  if (thread_num <= 4) {
721  pipeline();
722  ios.run();
723  write_all_files();
724  } else {
725  boost::asio::post(ios, pipeline);
726  boost::asio::post(ios, write_all_files);
727  ios.run();
728  // pipeline();
729  // ios.run();
730  // write_all_files();
731  }
732  spdlog::info("Done. Start clear temporary directory");
733  std::filesystem::remove_all(fwd_tmp_dir);
734  std::filesystem::remove_all(rev_tmp_dir);
735  }
