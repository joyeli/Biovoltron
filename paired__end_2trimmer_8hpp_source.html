158 auto find_possible_adapter_pos(SeqView seq,
159 SeqView adapter) const noexcept
160 -> std::vector<std::size_t> {
161 auto possible_pos = std::vector<std::size_t> {};
162
163 const auto read_size = std::min(detail::BASE_IN_VECTOR, seq.size());
164 const auto cmp_len = std::min(param.RC_CHECK_LEN, read_size);
165 const auto subseq = seq.substr(0, read_size);
166 auto seq_vec = detail::make_simd_vector(subseq, 1)[0];
167 const auto adapter_vec = detail::make_simd_vector(adapter, 1)[0];
168
169 /* interate all possible subsequence */
170 for (auto i = 0U; i <= read_size - cmp_len; i++) {
171 auto simd_sim = detail::cal_similarity(seq_vec, adapter_vec, cmp_len);
172 if (simd_sim >= param.TAIL_MATCH_RATIO) {
173 possible_pos.emplace_back(i + cmp_len);
174 }
175 seq_vec = detail::shift_left(seq_vec);
176 }
183 if (seq.size() > detail::BASE_IN_VECTOR) {
184 auto base_pos = detail::BASE_IN_VECTOR - param.RC_CHECK_LEN + 1;
185 auto&& r = find_possible_adapter_pos(seq.substr(base_pos), adapter);
186 std::ranges::transform(r, std::back_inserter(possible_pos),
187 [&](const auto& idx) {
188 return idx + base_pos;
189 });
190 }
191 return possible_pos;
192 }

208 auto get_adapter_pos(const SeqView forward_seq, const SeqView reverse_seq,
209 const std::vector<std::size_t>& intersect_pos)
210 const noexcept
211 -> std::size_t {
212
213 auto forward_rc = Codec::rev_comp(forward_seq);
214
215 /* we only need to make simd vector of reverse read for once */
216 auto rev_vec = detail::make_simd_vector(reverse_seq);
217
218 for (auto fwd_rc_view = SeqView(forward_rc);
219 auto pos : intersect_pos | std::views::reverse) {
220
221 auto fwd_rc_subseq = fwd_rc_view.substr(fwd_rc_view.size() - pos);
222 auto fwd_rc_subseq_vec = detail::make_simd_vector(fwd_rc_subseq);
223 auto match = 0;
224 for (auto i = 0u; i < fwd_rc_subseq_vec.size(); i++) {
225 auto cmp_len = std::min(
226 detail::BASE_IN_VECTOR,
227 pos - i * detail::BASE_IN_VECTOR
228 );
229 match += detail::cal_match(fwd_rc_subseq_vec[i], rev_vec[i], cmp_len);
230 }
231 if (static_cast<double>(match) / pos >= param.DNA_MATCH_RATIO) {
232 return pos;
233 }
234 }
235 return static_cast<std::size_t>(-1);
236 }

260 auto get_trim_pos(const SeqView forward_seq, const SeqView reverse_seq,
261 const std::vector<std::size_t>& intersect_pos,
262 const SeqView forward_adapter,
263 const SeqView reverse_adapter)
264 const noexcept -> std::size_t {
265
266 auto forward_rc = Codec::rev_comp(forward_seq);
267
268 /* we only need to make simd vector of reverse read for once */
269 auto rev_vec = detail::make_simd_vector(reverse_seq);
270 auto fwd_adapter_vec = detail::make_simd_vector(forward_adapter, 1)[0];
271 auto rev_adapter_vec = detail::make_simd_vector(reverse_adapter, 1)[0];
272
273 /* like `get_adapter_pos()`, we perform SS check here */
274 for (auto fwd_rc_view = SeqView(forward_rc);
275 auto pos : intersect_pos | std::views::reverse) {
276 if (pos == forward_seq.size()) {
277 continue;
278 }
279 auto fwd_rc_subseq = fwd_rc_view.substr(forward_rc.size() - pos);
280 auto fwd_rc_subseq_vec = detail::make_simd_vector(fwd_rc_subseq);
281
282 auto match = 0;
283 for (auto i = 0u; i < fwd_rc_subseq_vec.size(); i++) {
284 auto cmp_len = std::min(
285 detail::BASE_IN_VECTOR,
286 pos - i * detail::BASE_IN_VECTOR
287 );
288 match += detail::cal_match(fwd_rc_subseq_vec[i], rev_vec[i], cmp_len);
289 }
290
291 /* pass the SS check, perform AS check */
292 if (static_cast<double>(match) / pos >= param.DNA_MATCH_RATIO) {
293 auto fwd_subseq = forward_seq.substr(pos);
294 auto rev_subseq = reverse_seq.substr(pos);
295
296 auto fwd_subseq_vec = detail::make_simd_vector(fwd_subseq, 1)[0];
297 auto rev_subseq_vec = detail::make_simd_vector(rev_subseq, 1)[0];
298 auto len = std::min(fwd_subseq.size(), forward_adapter.size());
299
305 if (detail::cal_similarity(fwd_subseq_vec, fwd_adapter_vec, len)
306 >= param.ADAPTER_MATCH_RATIO ||
307 detail::cal_similarity(rev_subseq_vec, rev_adapter_vec, len)
308 >= param.ADAPTER_MATCH_RATIO) {
309 return pos;
310 }
311 }
312 }
313
318 if (intersect_pos.empty()) {
319 auto find_head_adapter_pos = [&](const SeqView seq,
320 const detail::SIMD_VECTOR& adapter_vec,
321 const std::size_t adapter_size) {
322 auto seq_vec = detail::make_simd_vector(seq, 1)[0];
323 auto cmp_len = std::min(param.HEAD_ADAPTER_CHECK_LEN, adapter_size);
324 for (auto pos : std::views::iota(0UL, param.HEAD_ADAPTER_CHECK_LEN)) {
325 cmp_len = std::min(seq.size() - pos, cmp_len);
326 if (cmp_len < param.MIN_ADAPTER_LEN) {
327 break;
328 }
329 auto similarity = detail::cal_similarity(
330 seq_vec, adapter_vec, cmp_len
331 );
332 if (similarity >= param.ADAPTER_MATCH_RATIO) {
333 return pos;
334 }
335 seq_vec = detail::shift_left(seq_vec);
336 }
337 return static_cast<std::size_t>(-1);
338 };
339
340 // ? figure out the origin implementation look like this
341 // from 32 to min_adapter_len, check the similarity of both sequence,
342 // check that whether adapter is at the head, then trim it.
343 {
344 auto forward_subseq = forward_seq.substr(
345 0, 2 * param.HEAD_ADAPTER_CHECK_LEN);
346 auto pos = find_head_adapter_pos(forward_subseq, fwd_adapter_vec,
347 forward_adapter.size());
348 if (pos != static_cast<std::size_t>(-1)) {
349 return pos;
350 }
351 }
352 {
353 auto reverse_subseq = reverse_seq.substr(
354 0, 2 * param.HEAD_ADAPTER_CHECK_LEN);
355 auto pos = find_head_adapter_pos(reverse_subseq, rev_adapter_vec,
356 reverse_adapter.size());
357 if (pos != static_cast<std::size_t>(-1)) {
358 return pos;
359 }
360 }
361 }
362
363 return forward_seq.size();
364 }

374 auto preprocess(R& fwd_read, R& rev_read) const noexcept -> void {
375 auto sz = std::min(fwd_read.seq.size(), rev_read.seq.size());
376 fwd_read.seq.resize(sz);
377 rev_read.seq.resize(sz);
378 if constexpr (std::same_as<R, FastqRecord<R::encoded>>) {
379 fwd_read.qual.resize(sz);
380 fwd_read.qual.resize(sz);
381 }
382 static std::mt19937 rng(
383 std::chrono::steady_clock::now().time_since_epoch().count()
384 );
385 auto check_for_N = [](auto c) {
386 if constexpr (R::encoded) {
387 if (c > 3) {
388 c = rng() & 3;
389 }
390 } else {
391 if (c == 'N') {
392 c = Codec::to_char(rng() & 3);
393 }
394 }
395 return c;
396 };
397 std::ranges::transform(fwd_read.seq, fwd_read.seq.begin(), check_for_N);
398 std::ranges::transform(rev_read.seq, rev_read.seq.begin(), check_for_N);

410 auto get_possible_intersect(SeqView fwd_seq,
411 SeqView rev_seq) const noexcept
412 -> std::vector<std::size_t> {
413 auto seq_pair = std::array { fwd_seq, rev_seq };
414 auto possible_pos = std::array<std::vector<std::size_t>, 2>{};
415
416 for (auto s = 0; s <= 1; s++) {
417 auto rc = Codec::rev_comp(seq_pair[s].substr(0, param.RC_CHECK_LEN));
418 possible_pos[s] = find_possible_adapter_pos(seq_pair[s ^ 1], rc);
419 }
420
421 auto intersect_pos = std::vector<std::size_t> {};
422 std::ranges::set_intersection(possible_pos[0],
423 possible_pos[1],
424 std::back_inserter(intersect_pos));
425 return intersect_pos;
426 };

436 auto detect_adapter(const std::vector<R>& forward_reads,
437 const std::vector<R>& reverse_reads,
438 bool sensitive) noexcept
439 -> std::pair<SeqType, SeqType> {
440
441 auto forward_tails = std::vector<SeqView> {};
442 auto reverse_tails = std::vector<SeqView> {};
443 auto sz = forward_reads.size();
444
445 spdlog::info("Collect tails from forward reads and reverse reads...");
446 for (auto i = 0u; i < sz; i++) {
447 auto fwd_seq = SeqView(forward_reads[i].seq);
448 auto rev_seq = SeqView(reverse_reads[i].seq);
449 auto intersect_pos = get_possible_intersect(fwd_seq, rev_seq);
450
451 /* No intersect positions, skip this reads */
452 if (intersect_pos.size() == 0) {
453 continue;
454 }
455
456 auto pos = get_adapter_pos(fwd_seq, rev_seq, intersect_pos);
457
458 /* cannot find proper adapter position */
459 if (pos == static_cast<std::size_t>(-1) || pos == fwd_seq.size()) {
460 continue;
461 }
462
463 forward_tails.push_back(fwd_seq.substr(pos));
464 reverse_tails.push_back(rev_seq.substr(pos));
465 }
466 spdlog::info("Done. Start detect adapter from collected tails");
467 SPDLOG_DEBUG("forward tails size = {}", forward_tails.size());
468 SPDLOG_DEBUG("reverse tails size = {}", reverse_tails.size());
469
470 spdlog::info("Detect forward adapter...");
471 auto forward_adapter = assembler.assemble(forward_tails, sensitive);
472 spdlog::info("Detect reverse adapter...");
473 auto reverse_adapter = assembler.assemble(reverse_tails, sensitive);
474
475 /* unable to assemble adapter, use default adapter */
476 if (forward_adapter.empty() || reverse_adapter.empty()) {
477 spdlog::info("Unable to detect adapter in reads");
478 spdlog::info("Use default adapter");
479 if constexpr (R::encoded) {
480 forward_adapter = Codec::to_istring(param.DEFAULT_ADAPTER1);
481 reverse_adapter = Codec::to_istring(param.DEFAULT_ADAPTER2);
482 } else {
483 forward_adapter = param.DEFAULT_ADAPTER1;
484 reverse_adapter = param.DEFAULT_ADAPTER2;
485 }
486 }
487
488 /* trim the adapter to same length */
489 if (forward_adapter.size() > reverse_adapter.size()) {
490 forward_adapter.resize(reverse_adapter.size());
491 } else if (forward_adapter.size() < reverse_adapter.size()) {
492 reverse_adapter.resize(forward_adapter.size());
493 }
494
495 spdlog::info("The forward adapter = {}", forward_adapter);
496 spdlog::info("The reverse adapter = {}", reverse_adapter);
497
498 return std::make_pair(forward_adapter, reverse_adapter);
499 }

511 auto trim_read(R& forward_read, R& reverse_read,
512 const SeqType& forward_adapter, const SeqType& reverse_adapter)
513 const noexcept
514 -> void {
515 auto fwd_seq_view = SeqView(forward_read.seq);
516 auto rev_seq_view = SeqView(reverse_read.seq);
517
518 auto intersect_pos = get_possible_intersect(fwd_seq_view, rev_seq_view);
519
520 auto trim_pos = get_trim_pos(fwd_seq_view, rev_seq_view,
521 intersect_pos,
522 forward_adapter, reverse_adapter);
523
524 if (trim_pos < param.MIN_READ_LEN) {
525 trim_pos = 0;
526 }
527 forward_read.seq.resize(trim_pos);
528 reverse_read.seq.resize(trim_pos);
529 if constexpr (std::same_as<R, FastqRecord<R::encoded>>) {
530 forward_read.qual.resize(trim_pos);
531 reverse_read.qual.resize(trim_pos);
532 }
533 }

547 auto asio_trim(std::ifstream& forward_fin, std::ifstream& reverse_fin,
548 std::ofstream& forward_fout, std::ofstream& reverse_fout,
549 int thread_num, const bool sensitive) {
550
551 using BufType = std::vector<R>;
552
553 auto tmp_dir = std::filesystem::temp_directory_path();
554 auto fwd_tmp_dir = tmp_dir / "fwd";
555 auto rev_tmp_dir = tmp_dir / "rev";
556 std::filesystem::remove_all(fwd_tmp_dir);
557 std::filesystem::remove_all(rev_tmp_dir);
558 if (!std::filesystem::create_directory(fwd_tmp_dir) ||
559 !std::filesystem::create_directory(rev_tmp_dir)) {
560 SPDLOG_ERROR("Cannot create temporary directory {} and {}", fwd_tmp_dir.string(), rev_tmp_dir.string());
561 std::exit(1);
562 }
563 auto ios = boost::asio::io_context {};
564 auto guard = boost::asio::make_work_guard(ios);
565 auto total_batch = -1LL;
566 auto writed_batch = std::set<std::size_t> {};
567 auto writed_batch_mtx = std::mutex {};
568
569 std::vector<std::jthread> threads(thread_num - 1);
570 for (auto i = 0; i < thread_num - 1; i++) {
571 threads.emplace_back(std::jthread([&ios]() {
572 ios.run();
573 }));
574 }
575
576 SPDLOG_DEBUG("Start Trimming by using Asychronous IO");
577
578 auto write_all_files = [&]() {
579 auto batch = 0LL;
580 auto final_write = [](std::filesystem::path p, std::ofstream& fout) {
581 auto fin = std::ifstream(p, std::ios::binary);
582 assert(fin.is_open());
583 fout << fin.rdbuf() << std::flush;
584 fin.close();
585 std::filesystem::remove(p);
586 };
587 while (1) {
588 while (writed_batch.contains(batch)) {
589 auto fwd_path = fwd_tmp_dir / std::to_string(batch);
590 auto rev_path = rev_tmp_dir / std::to_string(batch);
591 // SPDLOG_DEBUG("[write all] write {}, {}", fwd_path, rev_path);
592 final_write(fwd_path, forward_fout);
593 final_write(rev_path, reverse_fout);
594 batch++;
595 }
596 if (static_cast<int>(writed_batch.size()) == total_batch) {
597 assert(total_batch == batch);
598 // SPDLOG_DEBUG("[write all] Write files finished, batch = {}", batch);
599 break;
600 }
601 std::this_thread::sleep_for(1ms);
602 }
603 };
604
605 auto asio_read = [&](const std::size_t read_count) {
606 auto fwd_buf = BufType();
607 auto rev_buf = BufType();
608 fwd_buf.reserve(read_count);
609 rev_buf.reserve(read_count);
610 bool fwd_eof = false, rev_eof = false;
611 for (auto i = 0u; i < read_count; i++) {
612 if (R forward_read; !(forward_fin >> forward_read)) {
613 fwd_eof = true;
614 } else {
615 fwd_buf.emplace_back(std::move(forward_read));
616 }
617 if (R reverse_read; !(reverse_fin >> reverse_read)) {
618 rev_eof = true;
619 } else {
620 rev_buf.emplace_back(std::move(reverse_read));
621 }
622 if (fwd_eof != rev_eof) {
623 SPDLOG_ERROR("The forward reads file and reverse reads files doesn't contain same amount reads");
624 exit(EXIT_FAILURE);
625 } else if (fwd_eof) {
626 return std::make_tuple(true, std::move(fwd_buf), std::move(rev_buf));
627 }
628 }
629 return std::make_tuple(false, std::move(fwd_buf), std::move(rev_buf));
630 };
631
632 auto asio_write = [&](const std::size_t batch,
633 const BufType& fwd_buf, const BufType& rev_buf) {
634 auto write_to_tmp = [](std::filesystem::path p, const BufType& buf) {
635 auto fout = std::ofstream(p, std::ios::binary);
636 if (!fout.is_open()) {
637 SPDLOG_ERROR("Cannot open tmp file {}", p.string());
638 exit(EXIT_FAILURE);
639 }
640 for (const auto& r : buf) {
641 fout << r << '\n';
642 }
643 };
644
645 auto fwd_path = fwd_tmp_dir / std::to_string(batch);
646 auto rev_path = rev_tmp_dir / std::to_string(batch);
647 write_to_tmp(fwd_path, fwd_buf);
648 write_to_tmp(rev_path, rev_buf);
649 auto lock = std::scoped_lock<std::mutex>(writed_batch_mtx);
650 writed_batch.insert(batch);
651 };
652
653 auto&& [_, fwd_reads, rev_reads] = asio_read(param.DETECT_READS);
654 auto [forward_adapter, reverse_adapter]
655 = detect_adapter(fwd_reads, rev_reads, sensitive);
656
657 auto trim_reads = [&](const std::size_t batch,
658 BufType& fwd_buf, BufType& rev_buf) {
659 auto sz = fwd_buf.size();
660 for (auto i = 0u; i < sz; i++) {
661 trim_read(fwd_buf[i], rev_buf[i], forward_adapter, reverse_adapter);
662 }
663 asio_write(batch, fwd_buf, rev_buf);
664 };
665
666 auto pipeline = [&]() {
667 auto batch = 0;
668 auto bg = std::chrono::high_resolution_clock::now();
669 while (true) {
670 auto&& [eof, fwd_buf, rev_buf] = asio_read(param.ASIO_BUFFER_SIZE);
671 if (fwd_buf.empty()) {
672 break;
673 }
674 // SPDLOG_DEBUG("[Pipeline] Read batch {} finished", batch);
675 boost::asio::post(ios, std::bind(trim_reads, batch,
676 std::move(fwd_buf),
677 std::move(rev_buf)));
678 batch++;
679 if (eof) {
680 forward_fin.close();
681 reverse_fin.close();
682 break;
683 }
684 }
685 auto ed = std::chrono::high_resolution_clock::now();
686 SPDLOG_DEBUG("[Asio trim] read files takes {} ms", (ed - bg) / 1ms);
687 total_batch = batch;
688
689 boost::asio::post(ios, [&guard]() { guard.reset(); });
690 };
691
692 forward_fin.seekg(0);
693 reverse_fin.seekg(0);
694
695 spdlog::info("Start trimming...");
696 if (thread_num <= 4) {
697 pipeline();
698 ios.run();
699 write_all_files();
700 } else {
701 boost::asio::post(ios, pipeline);
702 boost::asio::post(ios, write_all_files);
703 ios.run();
704 // pipeline();
705 // ios.run();
706 // write_all_files();
707 }
708 spdlog::info("Done. Start clear temporary directory");
709 std::filesystem::remove_all(fwd_tmp_dir);
710 std::filesystem::remove_all(rev_tmp_dir);
711 }
