39  uint8v move_r(const uint8v &v, uint8_t init = 0u) {
40  auto ret = uint8v{};

77  auto local_backtrace(auto rn, auto sn, auto &d, auto ans_i, auto ans_j) {
78  auto cigar = biovoltron::Cigar{};
79  auto push = [&cigar](int size, char op) {
80  if (cigar.size() and cigar.back().op == op)
81  cigar.back().size += size;
82  else
83  cigar.emplace_back(size, op);
84  };
85 
86  if (sn - ans_j > 0)
87  push(sn - ans_j, 'S');
88 
89  auto segn = (sn - 1 + SIMD_WIDTH) / SIMD_WIDTH;
90  auto ptr = (uint8_t*)d.data();
91  int state = 0; // 0: H, 1: E, 2: F
92  while (ans_i > 0 and ans_j > 0) {
93  int idx = (ans_i - 1) * segn * SIMD_WIDTH + (ans_j - 1) % segn * SIMD_WIDTH + (ans_j - 1) / segn;
94  if (state == 0) {
95  if ((ptr[idx] & 0b111) == 0) {
96  break;
97  } else if (ptr[idx] & MATCH_MASK) {
98  push(1, 'M');
99  ans_i--;
100  ans_j--;
101  } else if (ptr[idx] & DELETE_MASK) {
102  state = 1;
103  } else if (ptr[idx] & INSERT_MASK) {
104  state = 2;
105  }
106  } else if (state == 1) {
107  if (ptr[idx] & DELETE_EXT_MASK) {
108  push(1, 'D');
109  ans_i--;
110  } else {
111  push(1, 'D');
112  ans_i--;
113  state = 0;
114  }
115  } else if (state == 2) {
116  if (ptr[idx] & INSERT_EXT_MASK) {
117  push(1, 'I');
118  ans_j--;
119  } else {
120  push(1, 'I');
121  ans_j--;
122  state = 0;
123  }
124  }
125  }
126 
127  if (ans_j > 0)
128  push(ans_j, 'S');
129 
130  std::reverse(cigar.begin(), cigar.end());
131 
132  return std::tuple<biovoltron::Cigar, uint32_t, uint32_t>{cigar, ans_i, ans_j};
133  }

270  auto v = reduce_max(h);
271  if (ans < v) {
272  auto *p = (uint8_t*)&h;
273  for (int x = 0; x < SIMD_WIDTH; x++) {

292  nxtH[0] = max(nxtH[0], move_r(H[segn - 1], bias) + p[0] - vbias);

297  auto pref = f;
298  auto preh = h;
299

301  auto nxtf = max(f - vIGE, h - vIGO);
302  nxtH[j] = max(nxtH[j], nxtf);
303

305  d_mask = d_mask | ((nxtH[j] != vbias) & (nxtH[j] == nxtf) & vINSERT_MASK);
306  d_mask = d_mask | ((f - vIGE == nxtf) & vINSERT_EXT_MASK);
307  d.emplace_back(d_mask);
308 
309  h = nxtH[j];
310  f = nxtf;
311  }
312 
313  f = move_r(f, bias);
314  h = move_r(h, bias);
315 
316  if (not reduce_or((f - pref) | (h - preh)))
317  break;
318 
319  d.resize(d.size() - segn);
320  }
321 
322 
323  for (int j = 0; j < segn; j++) {
324  uint8v d_mask = d[d.size() - segn + j];
325  d_mask = d_mask | ((nxtH[j] != vbias) & (nxtE[j] == nxtH[j]) & vDELETE_MASK);
326  d_mask = d_mask | ((E[j] - vDGE == nxtE[j]) & vDELETE_EXT_MASK);
327  d[d.size() - segn + j] = d_mask;
328  }
329 
330  for (int j = 1; j < segn; j++)
331  d[d.size() - segn + j] = d[d.size() - segn + j] | ((H[j - 1] + p[j] - vbias == nxtH[j]) & vMATCH_MASK);
332  d[d.size() - segn] = d[d.size() - segn] | ((move_r(H[segn - 1], bias) + p[0] - vbias == nxtH[0]) & vMATCH_MASK);
333 
334  std::swap(H, nxtH);
335  std::swap(E, nxtE);
336 
337  for (int j = 0; j < segn; j++)
338  update_ans(H[j], i, j);
339  }
340 
341  auto [cigar, ref_begin, seq_begin] = local_backtrace(rn, qn, d, ans_i, ans_j);
342  return std::tuple{uint8_t(ans - bias), ref_begin, ans_i, seq_begin, ans_j, cigar};

357  constexpr bool QUE_BEG_GAP_IS_PENALIZED = 1;

364  return gap_size ? -GO - gap_size * GE : 0;

367  if (rn == 0) {

371  biovoltron::Cigar{std::to_string(qn) + 'I'}

373  } else {
374  return std::tuple{
375  0,
376  biovoltron::Cigar{std::to_string(qn) + 'I'}

378  }
379  }
380

383  return std::tuple{
384  gap_penalty(rn, DGO, DGE),

386  };
387  } else {

411  if constexpr (REF_BEG_GAP_IS_PENALIZED) {
412  H.assign(segn, vIGO);

414  } else {

416  }

418  auto nxtH = SIMDVec(segn);
419  auto nxtE = SIMDVec(segn);
420

495  }
496  }

501  for (size_t j = i; j < segn * SIMD_WIDTH and ptr < qn; j += SIMD_WIDTH) {
502  if constexpr (not REF_END_GAP_IS_PENALIZED)
503  update_ans(lastRowS, rn, i);
504  lastRowS += p[j] - IGO - IGE;
505  ptr++;
506  }
507  }
508 
509  update_ans(lastRowS, rn, qn);
510 
511  auto cigar = global_backtrace(rn, qn, d, ans_i, ans_j);
512  return std::tuple{ ans, cigar };
513  }
