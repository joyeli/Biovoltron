309 auto
310 split_read(istring_view read) const {
311 constexpr auto npos = istring_view::npos;
312 auto frags = std::vector<istring_view>{};
313 for (auto start = npos, i = 0ul; i <= read.size(); i++) {
314 if (i == read.size() || read[i] == 4) {
315 if (start != npos && i - start >= args.SEED_LEN)
316 frags.push_back(read.substr(start, i - start));
317 start = npos;
318 } else if (start == npos)
319 start = i;
320 }
321 return frags;
322 }

327 static auto
328 need_revert(istring_view read, istring_view ref) {
329 const auto max_mis_cnt = (read.size() + 4) / 5;
330 auto mis_cnt = 0;
331 for (auto i = 0; mis_cnt <= max_mis_cnt && i < read.size(); i++)
332 if (read[i] != ref[i] && read[i] != 4 && ref[i] != 4)
333 mis_cnt++;
334 return mis_cnt <= max_mis_cnt;

341 auto
342 set_cigar(Aln& aln, istring_view read, auto& profile) const {
343 SPDLOG_DEBUG("============== compute cigar ==============");
344
345 if (!aln.cigar.empty())
346 return;
347 if (aln.score == read.size()) {
348 aln.cigar = std::to_string(read.size()) + 'M';
349 SPDLOG_DEBUG("full score: {}", aln.cigar);
350 aln.align_len = read.size();
351 return;
352 }
353
354 const auto sw_pos = aln.ref_end - read.size() - args.EXTEND;
355 const auto subref
356 = istring_view{ref.seq}.substr(sw_pos, read.size() + args.EXTEND + 1);
357
358 if (profile.read == nullptr)
359 profile = SseSmithWaterman::get_profile(read);
360 auto sw
361 = SseSmithWaterman::align(profile, subref, true, true, args.SW_THRESHOLD);
362
363 const auto ref_beg = sw_pos + sw.ref_beg;
364 const auto ref_end = sw_pos + sw.ref_end;
365 aln.pos = ref_beg;
366 aln.score = sw.score;
367 aln.cigar = std::move(sw.cigar);
368 aln.align_len
369 = std::max(sw.ref_end - sw.ref_beg + 1, sw.read_end - sw.read_beg + 1);
370
371 SPDLOG_DEBUG("pos: {}", aln.pos);
372 SPDLOG_DEBUG("raw cigar: {}", aln.cigar);
373 }

398 static auto
399 get_score(istring_view read, istring_view ref, bool forward)
400 -> std::pair<std::uint16_t, Cigar> {
401 const auto full_score = read.size();
402
403 const auto read_beg = read.substr(0, 5);
404 const auto ref_beg = ref.substr(0, 5);
405 const auto read_end = read.substr(read.size() - 5);
406 const auto ref_end = ref.substr(ref.size() - 5);
407
408 read.remove_prefix(5);
409 ref.remove_prefix(5);
410 read.remove_suffix(5);
411 ref.remove_suffix(5);
412
413 constexpr auto only_one_mismatch = [](istring_view read, istring_view ref) {
414 const auto [in1, in2] = std::ranges::mismatch(read, ref);
415 return read.substr(in1 - read.begin() + 1)
416 == ref.substr(in2 - ref.begin() + 1);
417 };
418
419 if (read != ref) {
420 if (read_beg != ref_beg || read_end != ref_end)
421 return {};
422 if (!only_one_mismatch(read, ref))
423 return {};
424
425 return {full_score - 5, std::to_string(full_score) + 'M'};
426 }
427
428 if (read_beg == ref_beg && read_end == ref_end) {
429 return {full_score, std::to_string(full_score) + 'M'};
430 }
431
432 constexpr auto get_beg_score =
433 [](auto read_beg, auto ref_beg) -> std::pair<std::uint16_t, std::string> {
434 constexpr static auto cigars
435 = std::array{"5S", "4S1M", "3S2M", "2S3M", "1S4M"};
436 for (auto clip_idx = 4; clip_idx >= 0; clip_idx--) {
437 if (read_beg[clip_idx] != ref_beg[clip_idx]) {
438 auto cigar = cigars[4 - clip_idx];
439
440 return {4 - clip_idx, cigar};
441 }
442 }
443 return {5, "5M"};
444 };
445
446 constexpr auto get_end_score =
447 [](auto read_end, auto ref_end) -> std::pair<std::uint16_t, std::string> {
448 constexpr static auto cigars
449 = std::array{"5S", "1M4S", "2M3S", "3M2S", "4M1S"};
450 for (auto clip_idx = 0; clip_idx <= 4; clip_idx++) {
451 if (read_end[clip_idx] != ref_end[clip_idx]) {
452 auto cigar = cigars[clip_idx];
453 return {clip_idx, cigar};
454 }
455 }
456 return {5, "5M"};
457 };
458
459 auto score = read.size();
460 auto mid_cigar = std::to_string(score) + 'M';
461 const auto [end_score, end_cigar] = get_end_score(read_end, ref_end);
462 const auto [beg_score, beg_cigar] = get_beg_score(read_beg, ref_beg);
463 score += (beg_score + end_score);
464 if (score < full_score - 5)
465 return {};
466 auto cigar = static_cast<Cigar>(beg_cigar + mid_cigar + end_cigar);
467 cigar.compact();
468 return {score, std::move(cigar)};
469 }

508 auto
509 pairing2(std::vector<Aln> alns1, std::vector<Aln> alns2) const {
510 std::ranges::sort(alns1);
511 std::ranges::sort(alns2);
512 auto aln_pairs = std::vector<AlnPair>{};
513 auto begin = alns2.begin();
514 auto end = begin;
515 for (const auto& aln1 : alns1) {
516 while (begin != alns2.end() && begin->pos < aln1.pos - args.PAIR_DIST)
517 ++begin;
518 while (end != alns2.end() && end->pos < aln1.pos + args.PAIR_DIST) ++end;
519 for (auto it = begin; it != end; ++it) {
520 const auto& aln2 = *it;
521 if (aln1.forward != aln2.forward)
522 aln_pairs.emplace_back(aln1, aln2);
523 }
524 }
525 std::ranges::sort(aln_pairs, std::ranges::greater{}, &AlnPair::score);
526 return aln_pairs;
527 }

552 auto
553 get_spans(istring_view read) const {
554 auto seed_spans = std::vector<SeedSpan>{};
555 auto repeat_size = 0;
556 auto origin_read = read;
557 while (read.size() >= args.SEED_LEN) {
558 const auto seed = read.substr(read.size() - args.SEED_LEN);
559 const auto [begin, end, offset] = index.get_range(seed, 0);
560 const auto span = index.get_offsets(begin, end);
561 if (span.size() <= args.MAX_HIT_CNT) {
562 const auto cur_seed = seed.substr(offset);
563 SPDLOG_DEBUG("seed: {}({}) -> ({})", cur_seed, cur_seed.size(), span.size());
564 if (!span.empty())
565 seed_spans.emplace_back(seed.substr(offset), span);
566 read.remove_suffix(args.SEED_LEN - offset - args.SEED_OVERLAP);
567 } else {
568 assert(offset == 0);
569 const auto seed2 = read.substr(0, read.size() - args.SEED_LEN);
570 const auto [begin2, end2, offset2]
571 = index.get_range(seed2, begin, end, args.MAX_HIT_CNT);
572 const auto span2 = index.get_offsets(begin2, end2);
573 if (span2.size() <= args.MAX_HIT_CNT) {
574 const auto cur_seed = read.substr(offset2);
575 SPDLOG_DEBUG("seed: {}({}) -> ({})", cur_seed, cur_seed.size(), span2.size());
576 if (!span2.empty())
577 seed_spans.emplace_back(read.substr(offset2), span2);
578 read = read.substr(0, offset2 + args.SEED_OVERLAP);
579 } else {
580 assert(offset2 == 0);
581 SPDLOG_DEBUG("seed: {}({}) -> ({})". read, read.size(), span2.size());
582 SPDLOG_DEBUG("\n************ backtrace extend ************");
583
584 auto remain_seed = origin_read.substr(read.size());
585 auto equal_offset = read.size();
586 auto equal_span = span2;
587 while (!remain_seed.empty()) {
588 const auto equal_size
589 = std::min(static_cast<int>(remain_seed.size()),
590 static_cast<int>(std::log2(equal_span.size()) / 2));
591 const auto equal_seed = remain_seed.substr(0, equal_size);
592 equal_span = get_equal_span(equal_span, equal_offset, equal_seed);
593
594 SPDLOG_DEBUG("equal_seed: {}({}) -> ({})", equal_seed, equal_seed.size(), equal_span.size());
595 remain_seed.remove_prefix(equal_size);
596 equal_offset += equal_size;
597 if (equal_span.size() <= args.MAX_HIT_CNT) {
598 const auto extend_seed = origin_read.substr(0, equal_offset);
599 seed_spans.emplace_back(extend_seed, equal_span);
600
601 SPDLOG_DEBUG("seed: {}({}) -> ({})", extend_seed, extend_seed.size(), equal_span.size());
602
603 break;
604 }
605 }
606
607 repeat_size = read.size();
608 read = {};
609 break;
610 }
611 }
612 }
613 // recover
614 if (read.size() >= args.SEED_LEN - args.SEED_OVERLAP) {
615 const auto [begin, end, offset] = index.get_range(read, 0);
616 const auto span = index.get_offsets(begin, end);
617 if (span.size() <= args.MAX_HIT_CNT) {
618 SPDLOG_DEBUG("seed: {} -> ({})", read.substr(offset), span.size());
619 if (!span.empty())
620 seed_spans.emplace_back(read.substr(offset), span);
621 }
622 }
623 return std::pair{std::move(seed_spans), repeat_size};
624 }

629 auto
630 seeding_impl(istring_view read, bool forward) const {
631 const auto frags = split_read(read);
632 SPDLOG_DEBUG("\n************ read seeds ************");
633
634 auto seed_spans = std::vector<SeedSpan>{};
635 auto repeats = 0;
636 for (const auto frag : frags) {
637 const auto [spans, repeat_size] = get_spans(frag);
638 repeats += repeat_size;
639 for (const auto span : spans) seed_spans.push_back(span);
640 }
641 std::ranges::sort(seed_spans);
642
643 auto anchors = std::vector<Anchor>{};
644 for (const auto [seed, span] :
645 seed_spans | std::views::take(args.MAX_SEED_CNT)) {
646 const auto seed_pos = seed.data() - read.data();
647 for (const auto ref_pos : span)
648 anchors.emplace_back(ref_pos, seed_pos, seed.size(), forward);
649 }
650
651 auto chain_map = std::map<std::uint32_t, std::vector<Anchor>>{};
652 for (const auto anchor : anchors) {
653 const auto read_pos = anchor.ref_pos - anchor.seed_pos;
654 const auto lower = chain_map.lower_bound(read_pos - args.SEED_LEN);
655 const auto upper = chain_map.upper_bound(read_pos + args.SEED_LEN);
656 if (lower != upper) {
657 for (auto it = lower; it != upper; ++it) it->second.push_back(anchor);
658 continue;
659 }
660 chain_map[read_pos].push_back(anchor);
661 }
662
663 auto chains = std::vector<std::vector<Anchor>>{};
664 for (auto& chain : chain_map | std::views::values) {
665 std::ranges::sort(chain);
666 chains.push_back(std::move(chain));
667 }
668 return std::pair{std::move(chains), repeats};
669 }

674 auto
675 seeding(istring_view read, istring_view rread) const {
676 auto [chains, repeats] = seeding_impl(read, true);
677
678 SPDLOG_DEBUG("\n************ reverse ************");
679 std::stringstream ss;
680 ss << "rread: ";
681 for (auto i = 0; i < rread.size(); i++)
682 ss << (rread[i] != 4 ? Codec::to_char(rread[i]) : '|');
683 SPDLOG_DEBUG("{}", ss.str());
684
685 const auto [rchains, rrepeats] = seeding_impl(rread, false);
686 std::ranges::copy(rchains, std::back_inserter(chains));
687 std::ranges::sort(chains, std::ranges::greater{},
688 &std::vector<Anchor>::size);
689 return std::tuple{std::move(chains), repeats, rrepeats};
690 }

695 auto
696 exact_match(auto& chains, istring_view read, istring_view rread,
697 int find_cnt) const {
698 auto alns = std::vector<Aln>{};
699 auto sw_chains = std::vector<std::vector<Anchor>>{};
700 for (auto& chain : chains) {
701 const auto [ref_pos, seed_pos, seed_size, forward, repeat]
702 = chain.front();
703
704 auto read_pos = ref_pos - seed_pos;
705 const auto sw_read = forward ? read : rread;
706
707 SPDLOG_DEBUG("ref_pos: {}", ref_pos);
708 SPDLOG_DEBUG("read_pos: {}", read_pos);
709 SPDLOG_DEBUG("sw_read: {}", sw_read);
710 if (const auto [score, cigar] = get_score(
711 sw_read, istring_view{ref.seq}.substr(read_pos, read.size()),
712 forward);
713 score) {
714 if (const auto [size, op] = cigar.front(); op == 'S') {
715 read_pos += size;
716 }
717 alns.emplace_back(read_pos, score, 0, forward).cigar = cigar;
718 alns.back().find_cnt = find_cnt;
719 alns.back().align_len = cigar.ref_size();
720 } else {
721 sw_chains.push_back(std::move(chain));
722 }
723 }
724 if (sw_chains.size() > args.MAX_EM_CNT)
725 sw_chains.resize(args.MAX_EM_CNT);
726 return std::pair{std::move(alns), std::move(sw_chains)};
727 }

734 auto kmers = std::vector<std::uint32_t>{};

746 auto
747 find_kmers(const std::vector<std::uint32_t>& kmers, istring_view ref,
748 std::vector<bool>& table) const {
749 table.clear();
750 table.resize(1 << args.KMER_SIZE * 2);
751
752 for (auto i = 0; i < ref.size() - args.KMER_SIZE + 1; i++)
753 table[Codec::hash(ref.substr(i, args.KMER_SIZE))] = true;
754
755 auto find_cnt = 0;
756 for (const auto kmer : kmers) find_cnt += table[kmer];
757 return find_cnt;

763 auto
764 get_sw_alns(const auto& chains, int read_size,
765 const std::vector<std::uint32_t>& kmers,
766 const std::vector<std::uint32_t>& rkmers,

768 auto alns = std::vector<Aln>{};
769 for (const auto& chain : chains) {
770 const auto [ref_pos, seed_pos, seed_size, forward, repeat]
771 = chain.front();
772 const auto read_pos = ref_pos - seed_pos;
773 const auto front_pad
774 = seed_pos <= args.EXTEND / 2 ? seed_pos * 2 : args.EXTEND;
775 const auto sw_pos = read_pos - front_pad;
776 const auto subref
777 = istring_view{ref.seq}.substr(sw_pos, read_size + 2 * args.EXTEND);
778
779 const auto find_cnt = find_kmers(forward ? kmers : rkmers, subref, table);
780 if (find_cnt < min_find_cnt)
781 continue;
782 min_find_cnt = std::max(find_cnt - args.MAX_FIND_CNT_DIFF, min_find_cnt);
783
784 alns.emplace_back(sw_pos, 0, 0, forward, 0, 0, find_cnt);
785 }
786 std::ranges::sort(alns, std::ranges::greater{}, &Aln::find_cnt);
787 return std::pair{std::move(alns), min_find_cnt};
788 }

815 auto
816 extending(std::vector<Aln>& alns, const std::vector<Aln>& sw_alns,
817 istring_view read, istring_view rread) const {
818 auto profile = s_profile{}, rprofile = s_profile{};
819 for (auto min_score = args.SW_THRESHOLD; const auto& sw_aln : sw_alns) {
820 const auto subref = istring_view{ref.seq}.substr(
821 sw_aln.pos, read.size() + 2 * args.EXTEND);
822 auto sw = SseSmithWaterman::SWResult{};
823 if (sw_aln.forward) {
824 if (profile.read == nullptr)
825 profile = SseSmithWaterman::get_profile(read);
826 sw = SseSmithWaterman::align(profile, subref, false, false, min_score);
827 } else {
828 if (rprofile.read == nullptr)
829 rprofile = SseSmithWaterman::get_profile(rread);
830 sw = SseSmithWaterman::align(rprofile, subref, false, false, min_score);
831 }

834 if (score < min_score)
835 continue;

837 alns.emplace_back(ref_end - sw.read_end, score, sw.score2, sw_aln.forward,
838 sw.read_end, ref_end, sw_aln.find_cnt);
839
840 min_score = std::max(min_score, score - args.MAX_SW_DIFF);
841 }
842 return std::pair{std::move(profile), std::move(rprofile)};

857 auto
858 rescue(const std::vector<Aln>& alns1, const std::vector<Aln>& alns2,
859 istring_view read2, istring_view rread2, auto& profile2,
860 auto& rprofile2, const std::vector<std::uint32_t>& kmers2,
861 const std::vector<std::uint32_t>& rkmers2, std::vector<bool>& table,
862 int min_find_cnt) const {
863 const auto [opt_score, sub_score, sub_cnt]
864 = get_opt_subopt_count(alns1 | std::views::transform(&Aln::score));
865 const auto rescue_cnt = std::min(sub_cnt + 1, args.MAX_RESCUE_CNT);
866
867 SPDLOG_DEBUG("============== rescue count: {} ==============", rescue_cnt);
868
869 auto rescues = std::vector<Aln>{};
870 for (auto min_score = args.SW_THRESHOLD;
871 const auto& aln1 : alns1 | std::views::take(rescue_cnt)) {
872 const auto pos1 = aln1.pos;
873 if (std::ranges::any_of(
874 alns2,
875 [pos1, PAIR_DIST = this->args.PAIR_DIST](const auto pos2) {
876 return DIFF(pos1, pos2) <= PAIR_DIST;
877 },
878 &Aln::pos)) {
879 SPDLOG_DEBUG("pos: {} already seen.", pos1);
880 continue;
881 }
882
883 const auto forward1 = aln1.forward;
884 const auto sw_pos = forward1 ? pos1 - args.EXTEND : pos1 - args.PAIR_DIST;
885 const auto subref = istring_view{ref.seq}.substr(
886 sw_pos, args.EXTEND + read2.size() + args.PAIR_DIST);
887
888 const auto find_cnt
889 = find_kmers(forward1 ? rkmers2 : kmers2, subref, table);
890 if (find_cnt < min_find_cnt)
891 continue;
892 min_find_cnt = std::max(find_cnt - args.MAX_FIND_CNT_DIFF, min_find_cnt);
893
894 auto sw = SseSmithWaterman::SWResult{};
895 if (forward1) {
896 if (rprofile2.read == nullptr)
897 rprofile2 = SseSmithWaterman::get_profile(rread2);
898 sw
899 = SseSmithWaterman::align(rprofile2, subref, false, false, min_score);
900 } else {
901 if (profile2.read == nullptr)
902 profile2 = SseSmithWaterman::get_profile(read2);
903 sw = SseSmithWaterman::align(profile2, subref, false, false, min_score);
904 }
905
906 const auto score = static_cast<int>(sw.score);
907 if (score < min_score)
908 continue;
909 const auto ref_end = sw_pos + sw.ref_end;
910 const auto ref_pos = ref_end - sw.read_end;
911 SPDLOG_DEBUG("{{ ref pos: {}({}), score: {} }}", ref_pos, static_cast<std::int32_t>(ref_pos-sw_pos), score);
912
913 rescues
914 .emplace_back(ref_pos, score, sw.score2, !forward1, sw.read_end,
915 ref_end, find_cnt)
916 .rescued
917 = true;
918 min_score = std::max(min_score, score - args.MAX_SW_DIFF);
919 }
920
921 return rescues;
922 }

927 auto
928 shrink_sw_size(int em_size1, auto& sw_alns1, int em_size2,
929 auto& sw_alns2) const {
930 if (sw_alns1.size() > args.MAX_SW_CNT)
931 sw_alns1.resize(args.MAX_SW_CNT);
932 if (em_size1 > args.MAX_EM_CNT)
933 sw_alns1.clear();
934 if (sw_alns2.size() > args.MAX_SW_CNT)
935 sw_alns2.resize(args.MAX_SW_CNT);
936 if (em_size2 > args.MAX_EM_CNT)
937 sw_alns2.clear();
938
939 if (em_size1 == 0 || em_size2 == 0)
940 return;
941
942 const auto sw_size1 = static_cast<int>(sw_alns1.size());
943 const auto sw_size2 = static_cast<int>(sw_alns2.size());
944 const auto total_size1 = em_size1 + sw_size1;
945 const auto total_size2 = em_size2 + sw_size2;
946
947 const auto shrink_size = std::min(total_size1, total_size2);
948 if (shrink_size < total_size1) {
949 const auto final_sw_size = std::max(0, shrink_size - em_size1);
950 if (sw_alns1.size() > final_sw_size)
951 sw_alns1.resize(final_sw_size);
952 }
953 if (shrink_size < total_size2) {
954 const auto final_sw_size = std::max(0, shrink_size - em_size2);
955 if (sw_alns2.size() > final_sw_size)
956 sw_alns2.resize(final_sw_size);
957 }

973 auto
974 get_best_pair(const std::vector<Aln>& alns1, const std::vector<Aln>& alns2,
975 const std::vector<AlnPair>& aln_pairs, istring_view read1,
976 istring_view rread1, auto& profile1, auto& rprofile1,
977 istring_view read2, istring_view rread2, auto& profile2,
978 auto& rprofile2, float frac_rep1, float frac_rep2) const {
979 SPDLOG_DEBUG("************ pairing results ************");
980 print_paires(aln_pairs);
981
982 const auto [opt_score1, sub_score1, sub_cnt1]
983 = get_opt_subopt_count(alns1 | std::views::transform(&Aln::score));
984 const auto [opt_score2, sub_score2, sub_cnt2]
985 = get_opt_subopt_count(alns2 | std::views::transform(&Aln::score));
986 const auto [opt_score, sub_score, sub_cnt] = get_opt_subopt_count(
987 aln_pairs | std::views::transform(&AlnPair::score));
988
989 auto [aln1, aln2] = aln_pairs.front();
990 const auto score_un = opt_score1 + opt_score2 - args.PEN_UNPAIRED;
991 const auto success = (opt_score > score_un);
992 if (!success) {
993 aln1 = alns1.front();
994 aln2 = alns2.front();
995 }
996 if (aln1.forward)
997 set_cigar(aln1, read1, profile1);
998 else
999 set_cigar(aln1, rread1, rprofile1);
1000 if (aln2.forward)
1001 set_cigar(aln2, read2, profile2);
1002 else
1003 set_cigar(aln2, rread2, rprofile2);
1004 if (!success) {
1005 aln1.mapq = mem_approx_mapq_se({aln1.score, aln1.score2, sub_score1,
1006 aln1.align_len, sub_cnt1, frac_rep1});
1007 aln2.mapq = mem_approx_mapq_se({aln2.score, aln2.score2, sub_score2,
1008 aln2.align_len, sub_cnt2, frac_rep2});
1009 } else {
1010 const auto [mapq1, mapq2]
1011 = mem_mapq_pe({aln1.score, aln1.score2, sub_score1, aln1.align_len,
1012 sub_cnt1, frac_rep1},
1013 {aln2.score, aln2.score2, sub_score2, aln2.align_len,
1014 sub_cnt2, frac_rep2},
1015 score_un, opt_score, sub_score, sub_cnt);
1016 SPDLOG_DEBUG("(raw mapq1:{}, raw mapq2: {})", mapq1, mapq2);
1017 const auto pen_paired = insert_penalty(aln_pairs.front().dist());
1018 aln1.mapq = std::max(mapq1 - pen_paired, 0);
1019 aln2.mapq = std::max(mapq2 - pen_paired, 0);
1020 }
1021 aln1.sub_score = (aln1.score == opt_score1 ? sub_score1 : opt_score1);
1022 aln2.sub_score = (aln2.score == opt_score2 ? sub_score2 : opt_score2);
1023 return AlnPair{std::move(aln1), std::move(aln2)};
1024 }

1034 auto
1035 map(istring_view read1, istring_view rread1, istring_view read2,
1036 istring_view rread2) const -> AlnPair {
1037 SPDLOG_DEBUG("--------------- seeding read1 ---------------");
1038
1039 const auto [chains1, repeats1, rrepeats1] = seeding(read1, rread1);
1040
1041 SPDLOG_DEBUG("\n--------------- seeding read2 ---------------");
1042
1043 const auto [chains2, repeats2, rrepeats2] = seeding(read2, rread2);
1044
1045 const auto frac_rep1 = (repeats1 + rrepeats1) / (read1.size() * 2.f);
1046 const auto frac_rep2 = (repeats2 + rrepeats2) / (read2.size() * 2.f);
1047
1048 auto table = std::vector<bool>(1 << args.KMER_SIZE * 2);
1049 const auto kmers1 = get_kmers(read1);
1050 const auto rkmers1 = get_kmers(rread1);
1051 const auto kmers2 = get_kmers(read2);
1052 const auto rkmers2 = get_kmers(rread2);
1053
1054 SPDLOG_DEBUG("\n--------------- exact match read1 ---------------");
1055
1056 auto [alns1, sw_chains1]
1057 = exact_match(chains1, read1, rread1, kmers1.size());
1058
1059 SPDLOG_DEBUG("\n--------------- exact match read2 ---------------");
1060
1061 auto [alns2, sw_chains2]
1062 = exact_match(chains2, read2, rread2, kmers2.size());
1063
1064 SPDLOG_DEBUG("\n--------------- sw read1 ---------------");
1065
1066 auto [sw_alns1, min_find_cnt1] = get_sw_candidates(
1067 alns1.empty(), sw_chains1, read1.size(), kmers1, rkmers1, table);
1068
1069 SPDLOG_DEBUG("\n--------------- sw read2 ---------------");
1070
1071 auto [sw_alns2, min_find_cnt2] = get_sw_candidates(
1072 alns2.empty(), sw_chains2, read2.size(), kmers2, rkmers2, table);
1073
1074 release_memory(sw_chains1);
1075 release_memory(sw_chains2);
1076
1077 shrink_sw_size(alns1.size(), sw_alns1, alns2.size(), sw_alns2);
1078
1079 auto [profile1, rprofile1] = extending(alns1, sw_alns1, read1, rread1);
1080 auto [profile2, rprofile2] = extending(alns2, sw_alns2, read2, rread2);
1081
1082 release_memory(sw_alns1);
1083 release_memory(sw_alns2);
1084
1085 if (alns1.empty() && alns2.empty()) [[unlikely]]
1086 return {};
1087
1088 finalize_alns(alns1);
1089 finalize_alns(alns2);
1090
1091 SPDLOG_DEBUG("\n************ force rescue read1 ************");
1092 auto rescues1 = rescue(alns2, alns1, read1, rread1, profile1, rprofile1,
1093 kmers1, rkmers1, table, min_find_cnt1);
1094 SPDLOG_DEBUG("\n************ force rescue read2 ************");
1095 auto rescues2 = rescue(alns1, alns2, read2, rread2, profile2, rprofile2,
1096 kmers2, rkmers2, table, min_find_cnt2);
1097
1098 if (!rescues1.empty()) {
1099 std::ranges::copy(rescues1, std::back_inserter(alns1));
1100 finalize_alns(alns1);
1101 }
1102 SPDLOG_DEBUG("\n************ read1 final result ({}) ************", alns1.size());
1103 print_alns(alns1);
1104
1105 if (!rescues2.empty()) {
1106 std::ranges::copy(rescues2, std::back_inserter(alns2));
1107 finalize_alns(alns2);
1108 }
1109 SPDLOG_DEBUG("\n************ read2 final result ({}) ************", alns2.size());
1110 print_alns(alns2);
1111
1112 release_memory(table);
1113 release_memory(rescues1);
1114 release_memory(rescues2);
1115
1116 if (alns2.empty())
1117 return {
1118 get_best_one(alns1, read1, rread1, profile1, rprofile1, frac_rep1), {}};
1119 if (alns1.empty())
1120 return {
1121 {}, get_best_one(alns2, read2, rread2, profile2, rprofile2, frac_rep2)};
1122
1123 SPDLOG_DEBUG("\n--------------- pairing ---------------");
1124
1125 auto aln_pairs = pairing2(alns1, alns2);
1126 if (aln_pairs.empty()) {
1127 SPDLOG_DEBUG("\n--------------- failed ---------------");
1128 return {
1129 get_best_one(alns1, read1, rread1, profile1, rprofile1, frac_rep1),
1130 get_best_one(alns2, read2, rread2, profile2, rprofile2, frac_rep2)};
1131 }
1132 return get_best_pair(alns1, alns2, aln_pairs, read1, rread1, profile1,
1133 rprofile1, read2, rread2, profile2, rprofile2,
1134 frac_rep1, frac_rep2);
1135 }
