174template<std::size_t N, std::unsigned_integral Block>
175class XbitIteratorBase {
176 public:
180 using iterator_category = std::random_access_iterator_tag;
184 using value_type = std::uint8_t;
188 using difference_type = std::ptrdiff_t;
192 using pointer = void;
196 constexpr static std::size_t xbits_per_block = sizeof(Block) * CHAR_BIT / N;
197
198 protected:
202 Block* seg_;
206 std::size_t offset_;
207
208 public:
215 constexpr XbitIteratorBase(Block* seg, std::size_t offset) noexcept
216 : seg_(seg), offset_(offset) { }
217
225 constexpr friend difference_type
226 operator-(const XbitIteratorBase& x, const XbitIteratorBase& y) {
227 return (x.seg_ - y.seg_) * xbits_per_block + x.offset_ - y.offset_;
228 }
229
236 constexpr bool
237 operator==(const XbitIteratorBase& other) const noexcept = default;
238
245 constexpr auto
246 operator<=>(const XbitIteratorBase& other) const noexcept {
247 if (auto cmp = seg_ <=> other.seg_; cmp != 0)
248 return cmp;
249 return offset_ <=> other.offset_;
250 }
251
252 protected:
258 constexpr void
259 bump_up() {
260 if (offset_ != xbits_per_block - 1)
261 ++offset_;
262 else {
263 offset_ = 0;
264 ++seg_;
265 }
266 }
267
273 constexpr void
274 bump_down() {
275 if (offset_ != 0)
276 --offset_;
277 else {
278 offset_ = xbits_per_block - 1;
279 --seg_;
280 }
281 }
282
288 constexpr void
289 incr(difference_type n) {
290 if (n >= 0)
291 seg_ += (n + offset_) / xbits_per_block;
292 else
293 seg_ += static_cast<difference_type>(n - xbits_per_block + offset_ + 1)
294 / static_cast<difference_type>(xbits_per_block);
295 n &= (xbits_per_block - 1);
296 offset_ = (n + offset_) & (xbits_per_block - 1);
297 }
298};
