15 enum {
16 M_TO_M,
17 M_TO_I,
18 M_TO_D,

21 D_TO_M,
22 D_TO_D,

26 using TrasMatrx = std::array<double, TRANS_PROB_ARRAY_LENGTH>;
27 static constexpr auto TRISTATE_CORRECTION = 3.0;
28 static constexpr auto MAXIMUM_BEST_ALT_LIKELIHOOD_DIFFERENCE = -4.5;
29 static constexpr auto EXPECTED_ERROR_RATE_PER_BASE = 0.02;
30 static constexpr auto LOG10_QUALITY_PER_BASE = -4.0;
31 static constexpr auto MAXIMUM_EXPECTED_ERROR_PER_READ = 2.0;
32 static constexpr auto ORIGINAL_DEFAULT
33 = TrasMatrx{0.9998, 0.0001, 0.0001, 0.9, 0.1, 0.9, 0.1};
34 static inline const auto INITIAL_CONDITION = std::pow(2, 1020);
35 static inline const auto INITIAL_CONDITION_LOG10
36 = std::log10(INITIAL_CONDITION);

38 private:
39 auto
40 initialize_priors(SamRecord<>& read, std::string_view haplotype,
41 auto& prior) const {
42 for (auto i = 0; i < read.size(); i++) {
43 const auto qual = read.qual[i];
44 const auto x = read.seq[i];
45 for (auto j = 0; j < haplotype.size(); j++) {
46 const auto y = haplotype[j];
47 prior[i + 1][j + 1]
48 = (x == y || x == 'N' || y == 'N' ?
49 1 - QualityUtils::qual_to_error_prob(qual) :
50 (QualityUtils::qual_to_error_prob(qual) / TRISTATE_CORRECTION));
51 }
52 }
53 }

55 auto
56 sub_compute_likelihood(SamRecord<>& read, std::string_view haplotype,
57 const TrasMatrx& trans, auto& M, auto& I, auto& D,
58 auto& prior) const {
59 auto initial_value = INITIAL_CONDITION / haplotype.size();
60 for (auto j = 0; j <= haplotype.size(); j++) D[0][j] = initial_value;
61
62 initialize_priors(read, haplotype, prior);
63
64 for (auto i = 1; i <= read.size(); i++) {
65 for (auto j = 1; j <= haplotype.size(); j++) {
66 M[i][j]
67 = prior[i][j]
68 * (M[i - 1][j - 1] * trans[M_TO_M] + I[i - 1][j - 1] * trans[I_TO_M]
69 + D[i - 1][j - 1] * trans[D_TO_M]);
70 I[i][j] = M[i - 1][j] * trans[M_TO_I] + I[i - 1][j] * trans[I_TO_I];
71 D[i][j] = M[i][j - 1] * trans[M_TO_D] + D[i][j - 1] * trans[D_TO_D];
72 }
73 }
74
75 auto final_sum_prob = 0.0;
76 const auto end_i = read.size();
77 for (auto j = 1; j <= haplotype.size(); j++)
78 final_sum_prob += M[end_i][j] + I[end_i][j];
79 return std::log10(final_sum_prob) - INITIAL_CONDITION_LOG10;
80 }

83 static auto
84 normalize_likelihoods(std::vector<std::vector<double>>& log_likelihoods) {
85 for (auto& likelihoods : log_likelihoods) {
86 auto best_likelihood = std::ranges::max(likelihoods);
87 auto cap_likelihood
88 = best_likelihood + MAXIMUM_BEST_ALT_LIKELIHOOD_DIFFERENCE;
89 for (auto& likelihood : likelihoods)
90 if (likelihood < cap_likelihood)
91 likelihood = cap_likelihood;
92 }
93 }

95 static auto
96 filter_poorly_modeled_reads(
97 std::vector<SamRecord<>>& reads,
98 std::vector<std::vector<double>>& log_likelihoods) {
99 auto filtered_reads = std::vector<SamRecord<>>{};
100 auto filtered_likelihoods = std::vector<std::vector<double>>{};
101 for (auto i = 0; i < log_likelihoods.size(); i++) {
102 auto best_likelihood = std::ranges::max(log_likelihoods[i]);
103 auto likelihood_threshold
104 = std::min(MAXIMUM_EXPECTED_ERROR_PER_READ,
105 std::ceil(reads[i].size() * EXPECTED_ERROR_RATE_PER_BASE))
106 * LOG10_QUALITY_PER_BASE;
107 if (best_likelihood < likelihood_threshold) {
108 filtered_reads.push_back(std::move(reads[i]));
109 filtered_likelihoods.push_back(std::move(log_likelihoods[i]));
110 }
111 }
112 reads.swap(filtered_reads);
113 log_likelihoods.swap(filtered_likelihoods);
114 }

116 auto
117 compute_likelihoods(const std::vector<Haplotype>& haplotypes,
118 std::vector<SamRecord<>>& reads,
119 const TrasMatrx& trans = ORIGINAL_DEFAULT) const {
120 auto get_padded_size = [](const auto& r) {
121 return 1 + std::ranges::max_element(r, {}, [](const auto& elem) {
122 return elem.size();
123 })->size();
124 };
125
126 auto M = std::vector(get_padded_size(reads),
127 std::vector(get_padded_size(haplotypes), 0.0));
128 auto I = M;
129 auto D = M;
130 auto prior = M;
131
132 auto log_likelihoods
133 = std::vector(reads.size(), std::vector(haplotypes.size(), 0.0));
134 for (auto i = 0; i < reads.size(); i++)
135 for (auto j = 0; j < haplotypes.size(); j++)
136 log_likelihoods[i][j] = sub_compute_likelihood(
137 reads[i], haplotypes[j].seq, trans, M, I, D, prior);
138 normalize_likelihoods(log_likelihoods);
139 filter_poorly_modeled_reads(reads, log_likelihoods);
140 return log_likelihoods;
141 }
