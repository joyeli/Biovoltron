36// Note: If map to reverse strand:
37// - reverse complement SEQ, TL
38// - reverse CIGAR, QUAL, MD
39inline auto aln_to_sam_list(const Alignment& aln) {
40 if (aln.hits.empty())
41 return std::vector<SamRecord<>>{};
42
43 auto tail_len = (aln.tail_pos == -1) ? 0 : aln.seq.size() - aln.tail_pos;
44 auto has_mismatch = !aln.hits.front().mismatches.empty();
45
46 auto sam = SamRecord{};
47 sam.qname = aln.name;
48 sam.flag = aln.forward ? 0 : 16;
49 sam.mapq = 255 - tail_len;
50 sam.cigar = (tail_len == 0)
51 ? std::to_string(aln.seq.size()) + 'M'
52 : std::to_string(aln.tail_pos) + 'M' + std::to_string(tail_len) + 'S';
53 if (!aln.forward) std::ranges::reverse(sam.cigar);
54 sam.rnext = "*";
55 sam.pnext = 0;
56 sam.tlen = 0;
57 sam.seq = aln.forward ? aln.seq : Codec::rev_comp(aln.seq);
58 sam.qual = aln.qual;
59 if (!aln.forward) std::ranges::reverse(sam.qual);
60
61 // NH tag: Number of reported alignments.
62 sam.optionals.emplace_back("NH:i:" + std::to_string(aln.hits.size()));
63
64 // Tail tag: Report tail.
65 if (tail_len != 0) {
66 auto tail = aln.forward ? aln.seq.substr(aln.tail_pos)
67 : Codec::rev_comp(aln.seq.substr(aln.tail_pos));
68 sam.optionals.emplace_back("TL:Z:" + tail);
69 }
70
71 // MD tag: Encoding mismatched and deleted reference bases.
72 // Used to reconstruct the mapped reference sequence without
73 // requiring access to the reference.
74 //
75 // [0-9]+(([A-Z]|\^[A-Z]+)[0-9]+)*
76 //
77 // ex: 10A5^AC6
78 // 10 matches followed by a mismatch with 'A' on the reference,
79 // and then 5 bases match and 2 base deletion ('AC' is the deltd::vector<Interval> alns{};
80 // std::vector<Mismatch> mismatches{};
81 // ted
82 // bases in the reference), and 6 more matches.
83 //
84 // Softclip (tail) doesn't count.
85
86 // Calculating match segments length between mismatches for MD tag.
87 auto match_segs = std::vector<std::uint32_t>{};
88 if (has_mismatch) {
89 std::ranges::transform(
90 aln.hits.front().mismatches,
91 std::back_inserter(match_segs),
92 [](const auto& mismatch) { return mismatch.pos; }
93 );
94 match_segs.push_back(aln.seq.size() - tail_len); // Exclude tail.
95 std::ranges::sort(match_segs);
96 std::adjacent_difference(
97 match_segs.begin(),
98 match_segs.end(),
99 match_segs.begin(),
100 [](auto a, auto b) { return a - b - 1; }
101 );
102 }
103
104 auto sams = std::vector<SamRecord<>>{aln.hits.size(), sam};
105
106 for (auto i = 0; i < aln.hits.size(); i++) {
107 sams[i].rname = aln.hits[i].intv.chrom;
108 sams[i].pos = aln.hits[i].intv.begin + 1; // 1-based coordinate.
109
110 if (has_mismatch) {
111 auto mismatches = aln.hits[i].mismatches;
112 std::ranges::sort(mismatches, {}, &Mismatch::pos);
113
114 auto md_str = std::string{};
115 for (auto i = 0; i < mismatches.size(); i++) {
116 md_str += std::to_string(match_segs[i])
117 + mismatches[i].correct_base;
118 }
119 md_str += std::to_string(match_segs.back());
120
121 if (!aln.forward) std::ranges::reverse(md_str);
122 sams[i].optionals.emplace_back("MD:Z:" + md_str);
123 }
124 }
125
126 return sams;
127}
