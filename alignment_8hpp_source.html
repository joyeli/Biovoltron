43 // - reverse complement SEQ, TL

45 inline auto aln_to_sam_list(const Alignment& aln) {
46  if (aln.hits.empty())
47  return std::vector<SamRecord<>>{};
48 
49  auto tail_len = (aln.tail_pos == -1) ? 0 : aln.seq.size() - aln.tail_pos;
50  auto has_mismatch = !aln.hits.front().mismatches.empty();
51 
52  auto sam = SamRecord{};
53  sam.qname = aln.name;
54  sam.flag = aln.forward ? 0 : 16;
55  sam.mapq = 255 - tail_len;
56  sam.cigar = (tail_len == 0)
57  ? std::to_string(aln.seq.size()) + 'M'
58  : std::to_string(aln.tail_pos) + 'M' + std::to_string(tail_len) + 'S';
59  if (!aln.forward) std::ranges::reverse(sam.cigar);
60  sam.rnext = "*";
61  sam.pnext = 0;
62  sam.tlen = 0;
63  sam.seq = aln.forward ? aln.seq : Codec::rev_comp(aln.seq);
64  sam.qual = aln.qual;
65  if (!aln.forward) std::ranges::reverse(sam.qual);
66 
67  // NH tag: Number of reported alignments.
68  sam.optionals.emplace_back("NH:i:" + std::to_string(aln.hits.size()));
69 
70  // Tail tag: Report tail.
71  if (tail_len != 0) {
72  auto tail = aln.forward ? aln.seq.substr(aln.tail_pos)
73  : Codec::rev_comp(aln.seq.substr(aln.tail_pos));
74  sam.optionals.emplace_back("TL:Z:" + tail);
75  }
76 
77  // MD tag: Encoding mismatched and deleted reference bases.
78  // Used to reconstruct the mapped reference sequence without
79  // requiring access to the reference.
80  //
81  // [0-9]+(([A-Z]|\^[A-Z]+)[0-9]+)*
82  //
83  // ex: 10A5^AC6
84  // 10 matches followed by a mismatch with 'A' on the reference,
85  // and then 5 bases match and 2 base deletion ('AC' is the deltd::vector<Interval> alns{};
86  // std::vector<Mismatch> mismatches{};
87  // ted
88  // bases in the reference), and 6 more matches.
89  //
90  // Softclip (tail) doesn't count.
91 
92  // Calculating match segments length between mismatches for MD tag.
93  auto match_segs = std::vector<std::uint32_t>{};
94  if (has_mismatch) {
95  std::ranges::transform(
96  aln.hits.front().mismatches,
97  std::back_inserter(match_segs),
98  [](const auto& mismatch) { return mismatch.pos; }
99  );
100  match_segs.push_back(aln.seq.size() - tail_len); // Exclude tail.
101  std::ranges::sort(match_segs);
102  std::adjacent_difference(
103  match_segs.begin(),
104  match_segs.end(),
105  match_segs.begin(),
106  [](auto a, auto b) { return a - b - 1; }
107  );
108  }
109 
110  auto sams = std::vector<SamRecord<>>{aln.hits.size(), sam};
111 
112  for (auto i = 0; i < aln.hits.size(); i++) {
113  sams[i].rname = aln.hits[i].intv.chrom;
114  sams[i].pos = aln.hits[i].intv.begin + 1; // 1-based coordinate.
115 
116  if (has_mismatch) {
117  auto mismatches = aln.hits[i].mismatches;
118  std::ranges::sort(mismatches, {}, &Mismatch::pos);
119 
120  auto md_str = std::string{};
121  for (auto i = 0; i < mismatches.size(); i++) {
122  md_str += std::to_string(match_segs[i])
123  + mismatches[i].correct_base;
124  }
125  md_str += std::to_string(match_segs.back());
126 
127  if (!aln.forward) std::ranges::reverse(md_str);
128  sams[i].optionals.emplace_back("MD:Z:" + md_str);
129  }
130  }
131 
132  return sams;
133 }
