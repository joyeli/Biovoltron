71 static auto
72 calculate_matrix(std::string_view ref, std::string_view alt,
73 std::vector<std::vector<int>>& score,
74 std::vector<std::vector<int>>& trace, Parameters params) {
75 const auto row_size = score.size();
76 const auto col_size = score[0].size();
77
78 // Gap tracking vectors for affine penalties
79 auto gap_size_down = std::vector(col_size + 1, 0);
80 auto best_gap_down
81 = std::vector(col_size + 1, std::numeric_limits<int>::min() / 2);
82 auto gap_size_right = std::vector(row_size + 1, 0);
83 auto best_gap_right
84 = std::vector(row_size + 1, std::numeric_limits<int>::min() / 2);
85
86 const auto [w_match, w_mismatch, w_open, w_extend] = params;
87 for (auto i = 1; i < row_size; i++) {
88 for (auto j = 1; j < col_size; j++) {
89 // Diagonal (match or mismatch)
90 const auto step_diag
91 = score[i - 1][j - 1]
92 + (ref[i - 1] == alt[j - 1] ? w_match : w_mismatch);
93
94 // Gap in ref (down)
95 const auto gap_open_down = score[i - 1][j] + w_open;
96 best_gap_down[j] += w_extend;
97 if (gap_open_down > best_gap_down[j]) {
98 best_gap_down[j] = gap_open_down;
99 gap_size_down[j] = 1;
100 } else
101 gap_size_down[j]++;
102 const auto step_down = best_gap_down[j];
103 const auto step_down_size = gap_size_down[j];
104
105 // Gap in alt (right)
106 const auto gap_open_right = score[i][j - 1] + w_open;
107 best_gap_right[i] += w_extend;
108 if (gap_open_right > best_gap_right[i]) {
109 best_gap_right[i] = gap_open_right;
110 gap_size_right[i] = 1;
111 } else
112 gap_size_right[i]++;
113 const auto step_right = best_gap_right[i];
114 const auto step_right_size = gap_size_right[i];
115
116 // Select the best move. The priority is diagonal > right > down.
117 if (step_diag >= step_down && step_diag >= step_right) {
118 score[i][j] = step_diag;
119 trace[i][j] = 0; // Diagonal move
120 } else if (step_right >= step_down) {
121 score[i][j] = step_right;
122 trace[i][j] = -step_right_size; // Insertion
123 } else {
124 score[i][j] = step_down;
125 trace[i][j] = step_down_size; // Deletion
126 }
127 }
128 }

138 static auto
139 calculate_cigar(std::vector<std::vector<int>>& score,
140 std::vector<std::vector<int>>& trace) {
141 const auto ref_size = score.size() - 1;
142 const auto alt_size = score.front().size() - 1;
143
144 auto max_score = std::numeric_limits<int>::min();
145 auto segment_len = 0;
146
147 // look for the largest score on the rightmost column. we use >= combined
148 // with the traversal direction to ensure that if two scores are equal, the
149 // one closer to diagonal gets picked
150 auto pos_i = 0;
151 for (auto i = 1; i <= ref_size; i++) {
152 const auto cur_score = score[i][alt_size];
153 if (cur_score >= max_score) {
154 max_score = cur_score;
155 pos_i = i;
156 }
157 }
158
159 // now look for a larger score on the bottom-most row
160 auto pos_j = alt_size;
161 auto diff = [](auto x, auto y) { return x > y ? x - y : y - x; };
162 for (auto j = 1; j <= alt_size; j++) {
163 const auto cur_score = score[ref_size][j];
164 if (cur_score > max_score
165 || (cur_score == max_score
166 && diff(ref_size, j) < diff(pos_i, pos_j))) {
167 max_score = cur_score;
168 pos_i = ref_size;
169 pos_j = j;
170 // end of alternate is overhanging; we will just record it as 'M'
171 // segment
172 segment_len = alt_size - j; // Soft clip overhangs
173 }
174 }
175
176 auto cigar = Cigar{};
177 if (segment_len > 0) {
178 cigar.emplace_back(segment_len, 'S'); // soft-clip at the end
179 segment_len = 0;
180 }
181
182 auto state = 'M'; // initial assumed state
183 do {
184 const auto cur_trace = trace[pos_i][pos_j];
185 const auto [new_state, step_size] = [cur_trace] {
186 if (cur_trace > 0)
187 return std::pair{'D', cur_trace};
188 else if (cur_trace < 0)
189 return std::pair{'I', -cur_trace};
190 else
191 return std::pair{'M', 1};
192 }();
193
194 // Update position based on move
195 switch (new_state) {
196 case 'M':
197 pos_i--;
198 pos_j--;
199 break;
200 case 'I':
201 pos_j -= step_size;
202 break;
203 case 'D':
204 pos_i -= step_size;
205 break;
206 default:
207 break;
208 }
209
210 // Build CIGAR operation
211 if (new_state == state)
212 segment_len += step_size;
213 else {
214 cigar.emplace_back(segment_len, state);
215 segment_len = step_size;
216 state = new_state;
217 }
218 } while (pos_i > 0 && pos_j > 0);
219
220 cigar.emplace_back(segment_len, state);
221 auto align_offset = pos_i;
222
223 if (pos_j > 0)
224 cigar.emplace_back(pos_j, 'S'); // soft-clip at the beginning
225
226 cigar.reverse();
227 return std::pair{align_offset, cigar};
228 }

243 static auto
244 align(std::string_view ref, std::string_view alt,
245 Parameters params = NEW_SW_PARAMETERS) {
246 assert(!ref.empty() && !alt.empty());
247
248 if (alt.size() == ref.size() && well_match(ref, alt))
249 return std::pair{0, Cigar(std::to_string(ref.size()) + 'M')};
250
251 auto score = std::vector(ref.size() + 1, std::vector(alt.size() + 1, 0));
252 auto trace = std::vector(ref.size() + 1, std::vector(alt.size() + 1, 0));
253 calculate_matrix(ref, alt, score, trace, params);
254 return calculate_cigar(score, trace);
255 }
