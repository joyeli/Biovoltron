40 static auto
41 calculate_matrix(std::string_view ref, std::string_view alt,
42 std::vector<std::vector<int>>& score,
43 std::vector<std::vector<int>>& trace, Parameters params) {
44 const auto row_size = score.size();
45 const auto col_size = score[0].size();
46
47 auto gap_size_down = std::vector(col_size + 1, 0);
48 auto best_gap_down
49 = std::vector(col_size + 1, std::numeric_limits<int>::min() / 2);
50 auto gap_size_right = std::vector(row_size + 1, 0);
51 auto best_gap_right
52 = std::vector(row_size + 1, std::numeric_limits<int>::min() / 2);
53
54 const auto [w_match, w_mismatch, w_open, w_extend] = params;
55 for (auto i = 1; i < row_size; i++) {
56 for (auto j = 1; j < col_size; j++) {
57 // step diag
58 const auto step_diag
59 = score[i - 1][j - 1]
60 + (ref[i - 1] == alt[j - 1] ? w_match : w_mismatch);
61
62 // step down
63 const auto gap_open_down = score[i - 1][j] + w_open;
64 best_gap_down[j] += w_extend;
65 if (gap_open_down > best_gap_down[j]) {
66 best_gap_down[j] = gap_open_down;
67 gap_size_down[j] = 1;
68 } else
69 gap_size_down[j]++;
70 const auto step_down = best_gap_down[j];
71 const auto step_down_size = gap_size_down[j];
72
73 // step right
74 const auto gap_open_right = score[i][j - 1] + w_open;
75 best_gap_right[i] += w_extend;
76 if (gap_open_right > best_gap_right[i]) {
77 best_gap_right[i] = gap_open_right;
78 gap_size_right[i] = 1;
79 } else
80 gap_size_right[i]++;
81 const auto step_right = best_gap_right[i];
82 const auto step_right_size = gap_size_right[i];
83
84 // priority here will be step diagonal, step down, step right
85 if (step_diag >= step_down && step_diag >= step_right) {
86 score[i][j] = step_diag;
87 trace[i][j] = 0;
88 } else if (step_right >= step_down) {
89 score[i][j] = step_right;
90 trace[i][j] = -step_right_size;
91 } else {
92 score[i][j] = step_down;
93 trace[i][j] = step_down_size;
94 }
95 }
96 }

99 static auto
100 calculate_cigar(std::vector<std::vector<int>>& score,
101 std::vector<std::vector<int>>& trace) {
102 const auto ref_size = score.size() - 1;
103 const auto alt_size = score.front().size() - 1;
104
105 auto max_score = std::numeric_limits<int>::min();
106 auto segment_len = 0;
107
108 // look for the largest score on the rightmost column. we use >= combined
109 // with the traversal direction to ensure that if two scores are equal, the
110 // one closer to diagonal gets picked
111 auto pos_i = 0;
112 for (auto i = 1; i <= ref_size; i++) {
113 const auto cur_score = score[i][alt_size];
114 if (cur_score >= max_score) {
115 max_score = cur_score;
116 pos_i = i;
117 }
118 }
119
120 // now look for a larger score on the bottom-most row
121 auto pos_j = alt_size;
122 auto diff = [](auto x, auto y) { return x > y ? x - y : y - x; };
123 for (auto j = 1; j <= alt_size; j++) {
124 const auto cur_score = score[ref_size][j];
125 if (cur_score > max_score
126 || (cur_score == max_score
127 && diff(ref_size, j) < diff(pos_i, pos_j))) {
128 max_score = cur_score;
129 pos_i = ref_size;
130 pos_j = j;
131 // end of alternate is overhanging; we will just record it as 'M'
132 // segment
133 segment_len = alt_size - j;
134 }
135 }
136
137 auto cigar = Cigar{};
138 if (segment_len > 0) {
139 cigar.emplace_back(segment_len, 'S');
140 segment_len = 0;
141 }
142
143 auto state = 'M';
144 do {
145 const auto cur_trace = trace[pos_i][pos_j];
146 const auto [new_state, step_size] = [cur_trace] {
147 if (cur_trace > 0)
148 return std::pair{'D', cur_trace};
149 else if (cur_trace < 0)
150 return std::pair{'I', -cur_trace};
151 else
152 return std::pair{'M', 1};
153 }();
154
155 // move to next best location in the sw matrix
156 switch (new_state) {
157 case 'M':
158 pos_i--;
159 pos_j--;
160 break;
161 case 'I':
162 pos_j -= step_size;
163 break;
164 case 'D':
165 pos_i -= step_size;
166 break;
167 default:
168 break;
169 }
170
171 if (new_state == state)
172 segment_len += step_size;
173 else {
174 cigar.emplace_back(segment_len, state);
175 segment_len = step_size;
176 state = new_state;
177 }
178 } while (pos_i > 0 && pos_j > 0);
179
180 cigar.emplace_back(segment_len, state);
181 auto align_offset = pos_i;
182 if (pos_j > 0)
183 cigar.emplace_back(pos_j, 'S');
184
185 cigar.reverse();
186 return std::pair{align_offset, cigar};
187 }

190 static auto
191 align(std::string_view ref, std::string_view alt,
192 Parameters params = NEW_SW_PARAMETERS) {
193 assert(!ref.empty() && !alt.empty());
194
195 if (alt.size() == ref.size() && well_match(ref, alt))
196 return std::pair{0, Cigar(std::to_string(ref.size()) + 'M')};
197
198 auto score = std::vector(ref.size() + 1, std::vector(alt.size() + 1, 0));
199 auto trace = std::vector(ref.size() + 1, std::vector(alt.size() + 1, 0));
200 calculate_matrix(ref, alt, score, trace, params);
201 return calculate_cigar(score, trace);
202 }
