58 r.qname = bam_get_qname(aln);
59 r.flag = aln->core.flag;
60 r.rname = aln->core.tid == -1 ? "*" : is.bam_header->target_name[aln->core.tid];
61 r.pos = aln->core.pos + 1;
62 r.mapq = aln->core.qual;
63 r.cigar.clear();
64 auto cigar = bam_get_cigar(aln);

76 auto seq = bam_get_seq(aln);
77 auto qual = bam_get_qual(aln);
78 r.seq.reserve(len);

509 template<bool Encoded>
510 auto aux_to_bin(bam1_t* aln, SamRecord<Encoded>& r) {
511 auto& len = aln->l_data;
512 auto aux_data = aln->data;
513 for (auto& opt : r.optionals) {
514 aux_data[len++] = opt[0];
515 aux_data[len++] = opt[1];
516 switch (opt[3]) {
517 case 'A': {
518 aux_data[len++] = 'A';
519 aux_data[len++] = opt[5];
520 break;
521 }
522 case 'i': {
523 if (opt[5] == '-') {
524 auto num = std::stoi(opt.substr(5));
525 if (num >= std::numeric_limits<std::int8_t>::min()) {
526 aux_data[len++] = 'c';
527 len += convert_type_and_copy<std::int8_t>(num, &aux_data[len]);
528 } else if (num >= std::numeric_limits<std::int16_t>::min()) {
529 aux_data[len++] = 's';
530 len += convert_type_and_copy<std::int16_t>(num, &aux_data[len]);
531 } else {
532 aux_data[len++] = 'i';
533 len += convert_type_and_copy<std::int32_t>(num, &aux_data[len]);
534 }
535 } else {
536 auto num = static_cast<uint32_t>(std::stoul(opt.substr(5)));
537 if (num <= std::numeric_limits<std::uint8_t>::max()) {
538 aux_data[len++] = 'C';
539 len += convert_type_and_copy<std::uint8_t>(num, &aux_data[len]);
540 } else if (num <= std::numeric_limits<std::uint16_t>::max()) {
541 aux_data[len++] = 'S';
542 len += convert_type_and_copy<std::uint16_t>(num, &aux_data[len]);
543 } else {
544 aux_data[len++] = 'I';
545 len += convert_type_and_copy<std::uint32_t>(num, &aux_data[len]);
546 }
547 }
548 break;
549 }
550 case 'f': {
551 aux_data[len++] = 'f';
552 auto num = std::stof(opt.substr(5));
553 std::memcpy(&aux_data[len], &num, sizeof(num));
554 len += sizeof(num);
555 break;
556 }
557 case 'Z':
558 case 'H': {
559 aux_data[len++] = opt[3];
560 std::ranges::copy_n(opt.substr(5).data(), opt.size() - 4, &aux_data[len]); // -4 = + 1 - 5
561 len += opt.size() - 4;
562 break;
563 }
564 case 'B': {
565 aux_data[len++] = 'B';
566 // subtype
567 aux_data[len++] = opt[5];
568 // preserve space for length
569 auto count = &aux_data[len];
570 len += sizeof(std::uint32_t);
571 // numbers
572 auto pos = 0ul;
573 auto arr_length = 0u;
574 // auto num_str = ""sv;
575 auto remaining = std::string_view(opt.begin() + 5, opt.end());
576 // auto remaining = static_cast<std::string_view>(opt.substr(5));
577 remaining = remaining.substr(2);
578 while (!remaining.empty()) {
579 pos = remaining.find(',');
580 // parse number
581 auto num_str = remaining.substr(0, pos);
582 if (opt[5] != 'f') {
583 auto num = std::stoll(std::string(num_str));
584 switch (opt[5]) {
585 case 'c': len += convert_type_and_copy<std::int8_t>(num, &aux_data[len]); break;
586 case 'C': len += convert_type_and_copy<std::uint8_t>(num, &aux_data[len]); break;
587 case 's': len += convert_type_and_copy<std::int16_t>(num, &aux_data[len]); break;
588 case 'S': len += convert_type_and_copy<std::uint16_t>(num, &aux_data[len]); break;
589 case 'i': len += convert_type_and_copy<std::int32_t>(num, &aux_data[len]); break;
590 case 'I': len += convert_type_and_copy<std::uint32_t>(num, &aux_data[len]);
591 }
592 } else {
593 auto num = std::stof(std::string(num_str));
594 std::memcpy(&aux_data[len], &num, sizeof(num));
595 len += sizeof(num);
596 }
597 // move to next number
598 if (pos != std::string::npos) remaining = remaining.substr(pos + 1);
599 else
600 remaining = remaining.substr(remaining.size());
601 ++arr_length;
602 }
603 std::memcpy(count, &arr_length, sizeof(arr_length));
604 }
605 }
606 }
607 aln->l_data = len;
608 }

611 template<bool Encoded = false>
612 auto bam_set1(
613 bam1_t* aln, SamRecord<Encoded>& r, int tid, std::vector<std::uint32_t>& cigars, std::vector<char>& quals,
614 int mtid) {
615 // determine data size
616 auto rlen = hts_pos_t { 0 };
617 auto qname_nuls = std::size_t { (4 - r.qname.size() % 4) % 4 };
618 if (!(r.flag & BAM_FUNMAP)) rlen = bam_cigar2rlen(cigars.size(), cigars.data());
619 if (rlen == 0) rlen = 1;
620 auto data_len = std::size_t {
621 r.qname.size() + qname_nuls + cigars.size() * 4 + (r.seq.size() + 1) / 2 +
622 r.seq.size() }; // read_name + digar + seq + qual
623 // realloc space
624 std::uint8_t* new_data;
625 auto new_m_data = data_len + 65535;
626 if ((bam_get_mempolicy(aln) & BAM_USER_OWNS_DATA) == 0)
627 new_data = reinterpret_cast<uint8_t*>(std::realloc(aln->data, new_m_data));
628 else {
629 if ((new_data = reinterpret_cast<uint8_t*>(std::malloc(new_m_data))) != nullptr) {
630 if (aln->l_data > 0)
631 std::ranges::copy(
632 aln->data, aln->data + (aln->l_data < aln->m_data ? aln->l_data : aln->m_data), new_data);
633 bam_set_mempolicy(aln, bam_get_mempolicy(aln) & (~BAM_USER_OWNS_DATA));
634 }
635 }
636 aln->data = new_data;
637 aln->m_data = new_m_data;
638
639 // set each field in aln structure
640 aln->l_data = data_len;
641 aln->core.pos = r.pos == 0 ? 0 : r.pos - 1;
642 aln->core.tid = tid;
643 aln->core.bin = hts_reg2bin(aln->core.pos, aln->core.pos + rlen, 14, 5);
644 aln->core.qual = r.mapq;
645 aln->core.l_extranul = qname_nuls - 1;
646 aln->core.flag = r.flag;
647 aln->core.l_qname = r.qname.size() + qname_nuls;
648 aln->core.n_cigar = cigars.size();
649 aln->core.l_qseq = r.seq.size();
650 aln->core.mtid = mtid;
651 aln->core.mpos = r.pnext == 0 ? -1ll : r.pnext - 1;
652 aln->core.isize = r.tlen;
653
654 // qname
655 auto data_ptr = aln->data;
656 std::ranges::copy(r.qname.begin(), r.qname.end(), data_ptr);
657 for (auto i = 0; i < qname_nuls; i++) data_ptr[r.qname.size() + i] = '\0';
658 data_ptr += r.qname.size() + qname_nuls;
659
660 // cigar
661 if (cigars.size() > 0) std::memcpy(data_ptr, cigars.data(), cigars.size() * 4);
662 data_ptr += cigars.size() * 4;
663
664 // seq
665 {
666 if (r.seq == "*") {
667 aln->core.l_qseq = 0;
668 } else {
669 for (auto i = 0u; i + 1 < r.seq.size(); i += 2) {
670 auto a = r.seq[i];
671 auto b = r.seq[i + 1];
672 if constexpr (Encoded) {
673 a = Codec::to_char(a);
674 b = Codec::to_char(b);
675 }
676 *data_ptr++ = (seq_nt16_table[a] << 4) | seq_nt16_table[b];
677 }
678 // odd length
679 if (r.seq.size() & 1) {
680 auto a = r.seq[r.seq.size() - 1];
681 if constexpr (Encoded) {
682 a = Codec::to_char(a);
683 }
684 *data_ptr++ = seq_nt16_table[a] << 4;
685 }
686 // for (; i + 1 < r.seq.size(); i += 2)
687 // *data_ptr++ = (seq_nt16_table[(unsigned char)r.seq[i]] << 4) |
688 // seq_nt16_table[(unsigned char)r.seq[i + 1]];
689 // for (; i < r.seq.size(); i++) *data_ptr++ = seq_nt16_table[(unsigned char)r.seq[i]] << 4;
690 }
691 }
692 // qual
693 if (r.qual != "*") {
694 std::ranges::copy(quals.begin(), quals.end(), data_ptr);
695 } else if (r.seq != "*") { // ? will this affected by Encoded
696 std::ranges::fill_n(data_ptr, quals.size(), '\xff');
697 }
698 // optional fields
699 aux_to_bin(aln, r);
700 }
