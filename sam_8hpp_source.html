97  static auto
98  compute_tlen(std::int32_t read_pos, const Cigar& read_cigar,
99  bool read_forward, std::int32_t mate_pos,
100  const Cigar& mate_cigar, bool mate_forward) noexcept
101  -> std::int32_t {
102  if (read_pos > mate_pos)
103  return -compute_tlen(mate_pos, mate_cigar, mate_forward, read_pos,
104  read_cigar, read_forward);
105 
106  switch (const auto read_ori = compute_ori(read_forward, mate_forward);
107  read_ori) {
108  case FR:
109  return mate_pos + mate_cigar.ref_size() - read_pos;
110  case FF:
111  if (const auto tlen = mate_pos + mate_cigar.read_size()
112  - (read_pos + read_cigar.read_size());
113  tlen != 0)
114  return tlen + (tlen > 0 ? 1 : -1);
115  else
116  return 0;
117  case RR:
118  if (const auto tlen = mate_pos + mate_cigar.ref_size()
119  - (read_pos + read_cigar.ref_size());
120  tlen != 0)
121  return tlen + (tlen > 0 ? 1 : -1);
122  else
123  return 0;
124  case RF:
125  if (const auto tlen = mate_pos - (read_pos + read_cigar.ref_size()) + 1;
126  tlen != 0)
127  return tlen + (tlen > 0 ? 1 : -1);
128  else
129  return 0;
130  default:
131  return 0;
132  }
