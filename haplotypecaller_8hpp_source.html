36 generate_reads_map(const std::vector<SamRecord<>>& sam) const {
37 std::vector<std::vector<SamRecord<>>> reads_map;
38 reads_map.resize(ref.seq.size());
39 for (const auto& record : sam)

41 reads_map[record.begin()].emplace_back(record);
42 return reads_map;

45 auto
46 sample_reads(const std::vector<SamRecord<>>& reads) const {
47 if (reads.size() <= args.MAX_READS_PER_ALIGN_BEGIN)
48 return reads;
49 auto sampled_reads = std::vector<SamRecord<>>{};
50 std::ranges::sample(reads, std::back_inserter(sampled_reads),
51 args.MAX_READS_PER_ALIGN_BEGIN,
52 std::mt19937{std::random_device{}()});
53 return sampled_reads;
54 }

56 static auto
57 filter_reads(std::vector<SamRecord<>>& reads) {
58 std::erase_if(reads, MappingQualityReadFilter{});
59 std::erase_if(reads, DuplicateReadFilter{});
60 std::erase_if(reads, SecondaryAlignmentReadFilter{});
61 std::erase_if(reads, MateOnSameContigReadFilter{});
62 }

64 static auto
65 hard_clip_reads(std::vector<SamRecord<>>& reads,
66 const Interval& padded_region) {
67 for (auto& read : reads) ReadClipper::revert_soft_clipped_bases(read);
68 for (auto& read : reads)
69 ReadClipper::hard_clip_to_interval(read, padded_region);
70 std::erase_if(reads, MinimumLengthReadFilter{});
71 }

73 auto
74 call_region(std::vector<SamRecord<>>& reads, std::string_view ref,
75 const Interval& padded_region,
76 const Interval& origin_region) const -> std::vector<Variant> {
77 filter_reads(reads);
78 hard_clip_reads(reads, padded_region);
79
80 if (reads.empty())
81 return {};
82 SPDLOG_DEBUG("----------------------------------------------------------------------------------");
83 SPDLOG_DEBUG("Assembling {} with {} reads: (with overlap region = {})", origin_region.to_string(),
84 reads.size(), padded_region.to_string());
85
86 auto haplotypes = assembler.assemble(reads, ref);
87 if (haplotypes.size() <= 1)
88 return {};
89
90 auto likelihoods = pairhmm.compute_likelihoods(haplotypes, reads);
91 SPDLOG_DEBUG("----------------------------------------------------------------------------------");
92 SPDLOG_DEBUG("Pairhmm values:");
93 std::stringstream ss;
94 for (auto i = 0; i < likelihoods.size(); i++) {
95 ss << "[" << i << "]: ";
96 for (auto j = 0; j < likelihoods[i].size(); j++)
97 ss << likelihoods[i][j] << " ";
98 SPDLOG_DEBUG("{}", ss.str());
99 ss.str("");
100 }
101
102 SPDLOG_DEBUG("----------------------------------------------------------------------------------");
103 SPDLOG_DEBUG("Genotyping:");
104 return genotyper.assign_genotype_likelihoods(
105 reads, haplotypes, likelihoods, ref, padded_region, origin_region);
106 }

109 auto
110 run(const std::vector<SamRecord<>>& sam) const {
111 const auto ref = static_cast<std::string_view>(this->ref.seq);
112 auto raw_variants = std::vector<VcfRecord>{};
113
114 // const auto start_begin = 16050000;
115 const auto start_begin = args.PADDING_SIZE;
116
117 auto window_cnt
118 = (ref.size() - start_begin + args.REGION_SIZE - 1) / args.REGION_SIZE;
119 auto origin_region
120 = Interval{this->ref.name, start_begin, start_begin + args.REGION_SIZE};
121 auto padded_region = origin_region;
122 padded_region.begin -= args.PADDING_SIZE;
123 padded_region.end += args.PADDING_SIZE;
124
125 auto reads_map = generate_reads_map(sam);
126 for (auto i = 0; i < window_cnt; i++) {
127 auto reads = std::vector<SamRecord<>>{};
128 for (auto begin = padded_region.begin; begin != padded_region.end;
129 begin++)
130 if (!reads_map[begin].empty() && begin < reads_map.size()) {
131 const auto sampled_reads = sample_reads(reads_map[begin]);
132 reads.insert(reads.end(), sampled_reads.begin(), sampled_reads.end());
133 }
134
135 if (reads.empty()) {
136 SPDLOG_DEBUG("Ignore {}: (with overlap region = {})", origin_region.to_string(), padded_region.to_string());
137 } else {
138 const auto region_variants = call_region(
139 reads, ref.substr(padded_region.begin, padded_region.size()),
140 padded_region, origin_region);
141 for (const auto& region_variant : region_variants)
142 SPDLOG_DEBUG(region_variant.to_string());
143 raw_variants.insert(raw_variants.end(), region_variants.begin(),
144 region_variants.end());
145 }
146
147 origin_region.begin += args.REGION_SIZE;
148 origin_region.end += args.REGION_SIZE;
149 padded_region.begin = origin_region.begin - args.PADDING_SIZE;
150 padded_region.end = origin_region.end + args.PADDING_SIZE;
151 }
152 SPDLOG_DEBUG("HaplotypeCaller done.");
153 return raw_variants;
154 }
