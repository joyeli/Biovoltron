24 constexpr auto match_in_byte = []() constexpr {
25  constexpr auto sz = (1u << (sizeof(std::uint8_t) * 8));
26  std::array<uint8_t, sz> hit { 0 };
27  std::ranges::for_each(std::views::iota(0u, sz), [&hit](auto idx) constexpr {

36 inline static std::vector<SIMD_VECTOR> erase_mask = []() {
37  auto mask = std::vector<SIMD_VECTOR>(BASE_IN_VECTOR + 1);

79 template<std::ranges::view SeqView>
80 auto make_simd_vector(SeqView seq, std::size_t vec_size = -1) {
81 
82  auto sz = seq.size();
83  auto need_size = sz / BASE_IN_VECTOR + (sz % BASE_IN_VECTOR != 0);
84  vec_size = std::min(vec_size, need_size);
85  auto vecs = std::vector<SIMD_VECTOR> {};
86 
87  auto fill_byte = [](SeqView s, std::uint8_t* ptr) {
88  *ptr = 0;
89  for (auto i = 0u; i < 4; i++) {
90  *ptr <<= 2;
91  if (i < s.size()) {
92  if constexpr (std::same_as<SeqView, istring_view>) {
93  *ptr |= s[i];
94  } else {
95  *ptr |= Codec::to_int(s[i]);
96  }
97  }
98  }
99  };
100 
101  for (auto i = 0u; i < vec_size; i++) {
102  auto ptr = static_cast<std::uint8_t*>(
103  std::aligned_alloc(VECTOR_SIZE, VECTOR_SIZE)
104  );
105  for (auto k = 0u; k < VECTOR_SIZE; k++) {
106  auto offset = i * BASE_IN_VECTOR + k * 4;
107  if (offset >= sz) {
108  break;
109  }
110  auto s = seq.substr(offset, 4);
111  fill_byte(s, ptr + k);
112  }
113  auto v = SIMD_VECTOR(simdpp::load(ptr));
114  vecs.push_back(v);
115  std::free(ptr);
116  }
117  return vecs;
118 };

120 auto shift_left(const SIMD_VECTOR& v) -> SIMD_VECTOR {
121  if constexpr (VECTOR_SIZE == 16) {
122  return v << 2 | simdpp::move16_l<1>(v >> 6);
123  } else {
124  /* simdpp::move16_l will perform each 128 bits seperately */
125 
126  /* how many 128 bits should be move left */
127  constexpr auto level = VECTOR_SIZE >> 4;
128 
129  /* store the 2 left most bits in v[8], v[16]... */
130  auto r = SIMD_VECTOR { v << 2 | simdpp::move16_l<1>(v >> 6) };
131 
132  /* restore the 2 left most bits in the right most bist of v[7], v[15]...*/
133  auto restore_bit = [&]<std::size_t... IDX>(std::index_sequence<IDX...>) {
134  (
135  [&]() {
136  constexpr auto pos = static_cast<uint8_t>((IDX + 1) << 4);
137  static_assert(0 < pos && pos < VECTOR_SIZE);
138  auto a = simdpp::extract<pos - 1>(r);
139  auto b = simdpp::extract<pos>(SIMD_VECTOR(v >> 6));
140  r = simdpp::insert<pos - 1>(r, a | b);
141  }(), ...
142  );
143  };
144  restore_bit(std::make_index_sequence<level - 1>());
145  return r;
146  }
