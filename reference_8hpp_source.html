209  std::conditional_t<Encoded, std::int8_t, char> ch {};
210  while (is >> ref) {
211  ++record.chr_num;
212  record.chr_end_pos.push_back( last_chr_pos + ref.seq.size() );
213  record.chr_names.push_back( ref.name );
214  for( auto i = std::uint32_t {0}; i < ref.seq.size(); ++i ) {
215  if( ref.seq[i] == 'N' or ref.seq[i] == 'n' or ref.seq[i] == 4) {
216  if( i == 0 or ref.seq[i] != ch ) {
217  record.unknown_intervals.push_back( {last_chr_pos + i, last_chr_pos + i + 1} );
218  }
219  else {
220  ++record.unknown_intervals.back()[1];
