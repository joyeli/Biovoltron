12/*
13 * A struct that store reference of a species
14 * which implemented using C++20
15 *
16 * sequence could be store as two types, DibitVector<std::uint8_t>, or std::string
17 * depend on the variable Encoded
18 *
19 * Example
20 * ```cpp
21 * #include <iostream>
22 * #include <experimental/random>
23 *
24 * using namespace biovoltron;
25 *
26 * int main(void) {
27 * auto ref = ReferenceRecord<false> { .species = "Human" };
28 * std::ifstream {"GRCh38"} >> ref;
29 *
30 * //chromosome name
31 * for(const auto &name : ref.chr_names) {
32 * std::cout << name << "\n";
33 * }
34 *
35 * //base count
36 * for(const auto &cnt : ref.base_cnt) {
37 * std::cout << cnt << " ";
38 * }
39 *
40 * //end position in each chromosome
41 * for(const auto &pos : ref.chr_end_pos) {
42 * std::cout << pos << " ";
43 * }
44 *
45 * //unknown intervals
46 * for(const auto &row : ref.unknown_intervals) [
47 * std::cout << row[0] << " - " << row[1] << "\n";
48 * }
49 *
50 * //save
51 * {
52 * auto fout = std::ostream{"GRCh38.bfa", std::ios::binary};
53 * ref.save(fout);
54 * fout.close();
55 * }
56 *
57 * //load
58 * auto load_ref = ReferenceRecord<false>{};
59 * {
60 * auto fin = std::ifstream{"GRCh38.bfa", std::ios::binary};
61 * load.ref.load(fin);
62 * fin.close();
63 * }
64 *
65 * assert(ref == load_ref);
66 *
67 * }
68 * ```
69 *
70 */

73template<bool Encoded = false>
74struct ReferenceRecord {
75 constexpr static auto encoded = Encoded;
76 std::string species {};
77 std::int32_t chr_num {};
78
79 std::vector<std::string> chr_names {};
80 std::conditional_t<Encoded, istring, std::string> seq {};
81
82 char (*substitute)(char&) = [](char &ch) { return (ch == 'N' or ch == 'n') ?
83 Codec::to_char(std::experimental::randint(0, 3)) : static_cast<char>(std::toupper(ch)); };
84
85 std::int8_t (*substitute_encoded)(std::int8_t&) = [](std::int8_t &ch) { return (ch > 3) ?
86 static_cast<std::int8_t>(std::experimental::randint(0, 3)) : ch; };
87
88 std::vector<std::uint32_t> base_cnt{}; //0:A, 1:C, 2:G, 3:T, 4:N
89 std::vector<std::uint32_t> chr_end_pos {};
90 std::vector<std::array<std::uint32_t, 2>> unknown_intervals {}; //[begin, end)
91
92 bool
93 operator==(const ReferenceRecord &other) const = default;

95 auto
96 origin_seq() {
97 std::conditional_t<Encoded, istring, std::string> o_seq {};
98 auto start = std::uint32_t {}, len = std::uint32_t{} ;
99 for( auto n_it = unknown_intervals.begin();
100 n_it != unknown_intervals.end(); ++n_it ) {
101 if( start <= (*n_it)[0] ) {
102 o_seq += seq.substr(start, (*n_it)[0] - start );
103 }
104 {
105 if constexpr ( encoded ) {
106 o_seq += Codec::to_istring(std::string( (*n_it)[1] - (*n_it)[0], 'N' ));
107 }
108 else {
109 o_seq += std::string( (*n_it)[1] - (*n_it)[0], 'N' );
110 }
111 }
112 start = (*n_it)[1];
113 }
114 o_seq += seq.substr(start);
115 return o_seq;
116 }

118 auto
119 save(std::ofstream &fout) const {
120 auto Dibit = DibitVector<std::uint8_t> {};
121 Dibit.reserve( seq.size() );
122 for( const auto &ch : seq ) {
123 if constexpr (encoded){
124 Dibit.push_back( ch );
125 }
126 else {
127 Dibit.push_back( Codec::to_int(ch) );
128 }
129 }
130 fout.write(reinterpret_cast<const char*>(&chr_num), sizeof(chr_num));
131 Serializer::save(fout, species);
132 Serializer::save(fout, Dibit);
133 Serializer::save(fout, base_cnt);
134 Serializer::save(fout, chr_end_pos);
135 Serializer::save(fout, unknown_intervals);
136
137 for(auto i {0}; i < chr_num; ++i) {
138 Serializer::save(fout, chr_names[i]);
139 }
140 }
141
142 auto
143 load(std::ifstream &fin) {
144 auto Dibit = DibitVector<std::uint8_t> {};
145 fin.read(reinterpret_cast<char*>(&chr_num), sizeof(chr_num));
146 Serializer::load(fin, species);
147 Serializer::load(fin, Dibit);
148 Serializer::load(fin, base_cnt);
149 Serializer::load(fin, chr_end_pos);
150 Serializer::load(fin, unknown_intervals);
151 chr_names = std::vector<std::string> (chr_num, "");
152 for(auto i {0}; i < chr_num; ++i) {
153 auto name = std::string {};
154 Serializer::load(fin, name);
155 chr_names[i] = name;
156 }
157 seq.reserve( Dibit.size() );
158 auto base_view = std::views::transform([](auto c){ if constexpr (encoded) return c; else return "ACGT"[c]; });
159 for( const auto &ch : Dibit | base_view ){
160 seq.push_back( ch );
161 }
162 }

166template<bool Encoded>
167inline auto&
168operator>>(std::istream &is, ReferenceRecord<Encoded> &record) {
169 record.base_cnt = std::vector<std::uint32_t> (5, 0);
170 auto ref = FastaRecord<Encoded>{};
171 auto last_chr_pos = std::uint32_t {0};
172 std::conditional_t<Encoded, std::int8_t, char> ch {};
173 while (is >> ref) {
174 ++record.chr_num;
175 record.chr_end_pos.push_back( last_chr_pos + ref.seq.size() );
176 record.chr_names.push_back( ref.name );
177 for( auto i = std::uint32_t {0}; i < ref.seq.size(); ++i ) {
178 if( ref.seq[i] == 'N' or ref.seq[i] == 'n' or ref.seq[i] == 4) {
179 if( i == 0 or ref.seq[i] != ch ) {
180 record.unknown_intervals.push_back( {last_chr_pos + i, last_chr_pos + i + 1} );
181 }
182 else {
183 ++record.unknown_intervals.back()[1];
184 }
185 ++record.base_cnt[4];
186 }
187 else { // not n
188 if constexpr (Encoded) {
189 ++record.base_cnt[ref.seq[i]];
190 }
191 else {
192 ++record.base_cnt[Codec::to_int(ref.seq[i])];
193 }
194 }
195 ch = ref.seq[i];
196 }
197 last_chr_pos = record.chr_end_pos.back();
198 if constexpr (Encoded) {
199 std::transform(ref.seq.begin(), ref.seq.end(), ref.seq.begin(), record.substitute_encoded );
200 }
201 else {
202 std::transform(ref.seq.begin(), ref.seq.end(), ref.seq.begin(), record.substitute );
203 }
204 //std::transform(ref.seq.begin(), ref.seq.end(), ref.seq.begin(),
205 // [](unsigned char c) { return std::toupper(c); } );
206
207 for(const auto &ch : ref.seq) {
208 record.seq.push_back(ch);
209 }
210
211 }
212 return is;
213}
