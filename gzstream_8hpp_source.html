44 class gzstreambuf : public std::streambuf {
45  private:
46  static const int bufferSize = 1 << 14; // size of data buff
47  // totals 512 bytes under g++ for igzstream at the end.
48 
49  gzFile file; // file handle for compressed file
50  char buffer[bufferSize]; // data buffer
51  char opened; // open/close state of stream
52  int mode; // I/O mode
53 
54  int
55  flush_buffer() {
56  // Separate the writing of the buffer from overflow() and
57  // sync() operation.
58  int w = pptr() - pbase();
59  if (gzwrite(file, pbase(), w) != w)
60  return EOF;
61  pbump(-w);
62  return w;
63  }
64 
65  public:
66  gzstreambuf*
67  open(const char* name, int open_mode) {
68  if (is_open())
69  return (gzstreambuf*)0;
70  mode = open_mode;
71  // no append nor read/write mode
72  if ((mode & std::ios::ate) || (mode & std::ios::app)
73  || ((mode & std::ios::in) && (mode & std::ios::out)))
74  return (gzstreambuf*)0;
75  char fmode[10];
76  char* fmodeptr = fmode;
77  if (mode & std::ios::in)
78  *fmodeptr++ = 'r';
79  else if (mode & std::ios::out)
80  *fmodeptr++ = 'w';
81  *fmodeptr++ = 'b';
82  *fmodeptr = '\0';
83  file = gzopen(name, fmode);
84  if (file == 0)
85  return (gzstreambuf*)0;
86  opened = 1;
87  return this;
88  }
89 
90  gzstreambuf*
91  close() {
92  if (is_open()) {
93  sync();
94  opened = 0;
95  if (gzclose(file) == Z_OK)
96  return this;
97  }
98  return (gzstreambuf*)0;
99  }
100 
101  virtual int
102  underflow() { // used for input buffer only
103  if (gptr() && (gptr() < egptr()))
104  return *reinterpret_cast<unsigned char*>(gptr());
105 
106  if (!(mode & std::ios::in) || !opened)
107  return EOF;
108  // Josuttis' implementation of inbuf
109  int num = gzread(file, buffer, bufferSize);
110  if (num <= 0) // ERROR or EOF
111  return EOF;
112 
113  // reset buffer pointers
114  setg(buffer, // beginning of putback area
115  buffer, // read position
116  buffer + num); // end of buffer
117 
118  // return next character
119  return *reinterpret_cast<unsigned char*>(gptr());
120  }
121 
122  virtual int
123  overflow(int c = EOF) { // used for output buffer only
124  if (!(mode & std::ios::out) || !opened)
125  return EOF;
126  if (c != EOF) {
127  *pptr() = c;
128  pbump(1);
129  }
130  if (flush_buffer() == EOF)
131  return EOF;
132  return c;
133  }
134 
135  virtual int
136  sync() {
137  // Changed to use flush_buffer() instead of overflow( EOF)
138  // which caused improper behavior with std::endl and flush(),
139  // bug reported by Vincent Ricard.
140  if (pptr() && pptr() > pbase()) {
141  if (flush_buffer() == EOF)
142  return -1;
143  }
144  return 0;
145  }
146 
147  gzstreambuf() : opened(0) {
148  setp(buffer, buffer + bufferSize);
149  setg(buffer, // beginning of putback area
150  buffer, // read position
151  buffer); // end position
152  // ASSERT: both input & output capabilities will not be used together
153  }
154  int
155  is_open() {
156  return opened;
157  }
158  ~gzstreambuf() { close(); }
159 };

161 class gzstreambase : virtual public std::ios {
162  protected:
163  gzstreambuf buf;
164 
165  public:
166  gzstreambase(const char* name, int mode) {
167  init(&buf);

169  }
170 
171  ~gzstreambase() { buf.close(); }
172 
173  void
174  open(const char* name, int open_mode) {
175  if (!buf.open(name, open_mode))
176  clear(rdstate() | std::ios::badbit);
177  }
178 
179  void
180  close() {
181  if (buf.is_open())
182  if (!buf.close())
183  clear(rdstate() | std::ios::badbit);
184  }
185  gzstreambase() { init(&buf); }
186  gzstreambuf*
187  rdbuf() {
