12  constexpr static auto SNP_PSEUDOCOUNT = 0.01;
13  constexpr static auto INDEL_PSEUDOCOUNT = 0.00125;
14  constexpr static auto THRESHOLD_FOR_ALLELE_COUNT_CONVERGENCE = 0.1;
15  constexpr static auto HOM_REF_GENOTYPE_INDEX = 0;
16  constexpr static auto SPAN_DEL = "*";

18  static auto
19  log10_normalized_genotype_posteriors(
20  std::span<const double> log10_genotype_likelihoods,
21  std::span<const double> log10_allele_frequencies,
22  std::span<const Genotype> genotypes) {
23  auto log10_posteriors = std::vector(genotypes.size(), 0.0);
24  for (auto i = 0; i < log10_posteriors.size(); i++) {
25  const auto [allele1, allele2] = genotypes[i];
26  const auto log10_combination_count
27  = allele1 == allele2 ? 0 : std::log10(2);
28  const auto sum
29  = log10_allele_frequencies[allele1] + log10_allele_frequencies[allele2];
30  log10_posteriors[i]
31  = log10_combination_count + log10_genotype_likelihoods[i] + sum;
32  }
33  return MathUtils::normalize_log10(log10_posteriors);
34  }

36  static auto
37  effective_allele_counts(std::span<const double> log10_genotype_likelihoods,
38  std::span<const double> log10_allele_frequencies,
39  std::span<const Genotype> genotypes) {
40  const auto log10_genotype_posteriors = log10_normalized_genotype_posteriors(
41  log10_genotype_likelihoods, log10_allele_frequencies, genotypes);
42 
43  auto log10_result = std::vector(log10_allele_frequencies.size(),
44  std::numeric_limits<double>::lowest());
45  for (auto i = 0; i < genotypes.size(); i++) {
46  const auto [allele1, allele2] = genotypes[i];
47  log10_result[allele1] = MathUtils::log10_sum_log10(
48  log10_result[allele1], log10_genotype_posteriors[i]);
49  log10_result[allele2] = MathUtils::log10_sum_log10(
50  log10_result[allele2], log10_genotype_posteriors[i]);
51  }
52  for (auto& x : log10_result) x = std::pow(10, x);
53  return log10_result;
54  }

66  static auto
67  calculate(std::span<const double> log10_genotype_likelihoods,
68  std::span<const std::string> alleles,
69  std::span<const Genotype> genotypes) {
70  const auto num_alleles = alleles.size();
71  auto prior_pseudocounts = std::vector(num_alleles, 0.0);
72  prior_pseudocounts.front() = REF_PSEUDOCOUNT;
73  for (auto i = 1; i < num_alleles; i++)
74  prior_pseudocounts[i] = alleles[i].size() == alleles[0].size() ?
75  SNP_PSEUDOCOUNT :
76  INDEL_PSEUDOCOUNT;
77 
78  auto log10_allele_frequencies
79  = std::vector(num_alleles, -std::log10(num_alleles));
80 
81  auto allele_counts = std::vector(num_alleles, 0.0);
82  for (auto allele_counts_maximum_difference
83  = std::numeric_limits<double>::max();
84  allele_counts_maximum_difference
85  > THRESHOLD_FOR_ALLELE_COUNT_CONVERGENCE;) {
86  const auto new_allele_counts = effective_allele_counts(
87  log10_genotype_likelihoods, log10_allele_frequencies, genotypes);
88 
89  const auto allele_counts_difference = RangeUtils::binary_transform(
90  allele_counts, new_allele_counts, std::minus{});
91  allele_counts_maximum_difference = std::ranges::max(
92  allele_counts_difference
93  | std::views::transform([](auto x) { return std::abs(x); }));
94 
95  allele_counts = new_allele_counts;
96  const auto posterior_pseudocounts = RangeUtils::binary_transform(
97  prior_pseudocounts, allele_counts, std::plus{});
98 
99  // first iteration uses flat prior in order to avoid local minimum where
100  // the prior + no pseudocounts gives such a low effective allele frequency
101  // that it overwhelms the genotype likelihood of a real variant basically,
102  // we want a chance to get non-zero pseudocounts before using a prior
103  // that's biased against a variant
104  log10_allele_frequencies
105  = MathUtils::dirichlet_log10_mean_weights(posterior_pseudocounts);
106  }
107 
108  const auto log10_genotype_posteriors = log10_normalized_genotype_posteriors(
109  log10_genotype_likelihoods, log10_allele_frequencies, genotypes);
110 
111  auto log10p_of_zero_counts_by_allele = std::vector(num_alleles, 0.0);
112  auto log10p_no_variant = 0.0;
113  auto log10_absent_posteriors
114  = std::vector<std::vector<double>>(num_alleles);
115  if (std::ranges::find(alleles, SPAN_DEL) == alleles.end())
116  log10p_no_variant += log10_genotype_posteriors[HOM_REF_GENOTYPE_INDEX];
117  else {
118  const auto non_variant_indices
119  = genotype_indices_with_only_ref_and_span_del(alleles, genotypes);
120 
121  auto non_variant_log10_posteriors = std::vector<double>{};
122  for (const auto n : non_variant_indices)
123  non_variant_log10_posteriors.push_back(log10_genotype_posteriors[n]);
124 
125  log10p_no_variant += std::min(
126  0.0, MathUtils::log10_sum_log10(non_variant_log10_posteriors));
127  }
128 
129  if (num_alleles != 2) {
130  for (auto i = 0; i < genotypes.size(); i++) {
131  const auto log10_genotype_posterior = log10_genotype_posteriors[i];
132  const auto [allele1, allele2] = genotypes[i];
133  for (auto n = 0; n < log10_absent_posteriors.size(); n++)
134  if (n != allele1 && n != allele2)
135  log10_absent_posteriors[n].push_back(log10_genotype_posterior);
136  }
137 
138  auto log10p_no_allele = std::vector<double>{};
139  for (const auto& buffer : log10_absent_posteriors)
140  log10p_no_allele.push_back(
141  std::min(0.0, MathUtils::log10_sum_log10(buffer)));
142 
143  for (auto i = 0; i < log10p_of_zero_counts_by_allele.size(); i++)
144  log10p_of_zero_counts_by_allele[i] += log10p_no_allele[i];
145  }
146 
147  if (num_alleles == 2)
148  log10p_of_zero_counts_by_allele[1] = log10p_no_variant;
149 
150  auto integer_allele_counts = std::vector<int>{};
151  for (const auto x : allele_counts)
152  integer_allele_counts.push_back(std::round(x));
153 
154  auto mle_counts_and_log10p_ref_by_allele
155  = std::vector<std::pair<std::string, std::pair<int, double>>>{};
156  for (auto i = 1; i < num_alleles; i++)
157  mle_counts_and_log10p_ref_by_allele.emplace_back(
158  alleles[i], std::pair{integer_allele_counts[i],
159  log10p_of_zero_counts_by_allele[i]});
160 
161  return std::pair{log10p_no_variant,
162  std::move(mle_counts_and_log10p_ref_by_allele)};
163  }

165  static auto
166  calculate_single_sample_biallelic_non_ref_posterior(
167  std::array<double, 3> log10_genotype_likelihoods) {
168  if (std::ranges::max_element(log10_genotype_likelihoods)
169  == log10_genotype_likelihoods.cbegin())
170  return 0.0;
171  auto log10_unnormalized_posteriors = std::array<double, 3>{};
172  for (auto i = 0; i < log10_unnormalized_posteriors.size(); i++) {
173  log10_unnormalized_posteriors[i]
174  = log10_genotype_likelihoods[i]
175  + MathUtils::log10_binomial_coefficient(2, i)
176  + MathUtils::log_to_log10(std::lgamma(i + SNP_PSEUDOCOUNT)
177  + std::lgamma(2 - i + REF_PSEUDOCOUNT));
178  }
179 
180  if (std::ranges::max_element(log10_unnormalized_posteriors)
181  == log10_unnormalized_posteriors.cbegin())
182  return 0.0;
183  return 1
184  - MathUtils::normalize_from_log10_to_linear_space(
185  log10_unnormalized_posteriors)[0];
186  }
