66 while (iter < MAX_ITERATIONS_TO_ATTEMPT) {

125 template<class SeqType>
126 auto low_complexity(const SeqType& seq) const -> bool {
127 auto cnt = std::array<int, 4>{};
128 for (auto& base : seq) {
129 if constexpr (std::same_as<SeqType, std::string>) {
130 cnt[Codec::to_int(base)] += 1;
131 } else {
132 cnt[base] += 1;
133 }
134 }
135 return std::ranges::any_of(cnt, [this, &seq](auto v) {
136 return v >= seq.size() * param.LOW_COMPLEXITY_RATE;
137 });
138 };

165 auto assemble(const std::vector<Seq>& seqs, bool sensitive = false) noexcept {
172 const auto max_iter =
173 sensitive ? param.SEN_MAX_ITERATIONS : param.MAX_ITERATIONS;
174 auto kmer_size = param.INIT_KMER_SIZE;
175 const auto max_kmer_size = param.MAX_KMER_SIZE;
176
177 auto iter = 1u;
178 auto is_low_complexity = false;
179 while (iter <= max_iter && kmer_size <= max_kmer_size) {
180
185 const auto pruning_factor = param.PRUNE_FACTOR / iter;
186 const auto minimum_occurence = std::max(
187 static_cast<std::size_t>(std::ceil(seqs.size() * pruning_factor)),
188 sensitive ? param.SEN_MINIMUM_OCCURENCE : param.MINIMUM_OCCURENCE
189 );
190 SPDLOG_DEBUG("Run iter {}", iter);
191 SPDLOG_DEBUG("kmer size = {}", kmer_size);
192 SPDLOG_DEBUG("Pruning factor = {}", pruning_factor);
193 SPDLOG_DEBUG("Minimum occurance = {}", minimum_occurence);
194
195 spdlog::info("Try to detect adapter with prune factor = {:.4f}", pruning_factor);
196 auto adapters = assemble(
197 seqs,
198 kmer_size,
199 minimum_occurence
200 );
201
202 /* cannot assemble proper adapters */
203 if (adapters.empty()) {
204 spdlog::info("Failed :(");
205 iter++;
206 continue;
207 }
208 auto& adapter = adapters[0];
209 /* the assembled adpater have low complexity */
210 if (low_complexity(adapter)) {
211 spdlog::info("The assembled adapter has low complexity");
212 is_low_complexity = true;
213 kmer_size += 5;
214 continue;
215 }
221 if (is_low_complexity) {
222 adapter.resize(std::min(adapter.size(), 15ul));
223 } else {
224 adapter.resize(std::min(adapter.size(), 32ul));
225 }
226 spdlog::info("Success: Assembled adapter = {}", adapter);
227 return adapter;
228 }
229 if constexpr (std::same_as<Seq, istring> || std::same_as<Seq, istring_view>) {
230 return istring{};
231 } else {
232 return std::string{};
233 }
234 }
