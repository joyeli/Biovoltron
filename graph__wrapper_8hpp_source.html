85 auto find_paths(Vertex from, Vertex to, bool use_edge_filter = true) {
86 auto paths = std::vector<Path>{};
87 auto path = Path{};
88 std::function<void(Vertex)> dfs = [&](Vertex u) {
89 path.push_back(u);
90 if (to == u) {
91 paths.push_back(path);
92 } else {
93 for (auto e : out_edges(u, use_edge_filter)) {
94 const auto v = target(e);
95 if (std::ranges::find(path, v) == path.end()) {
96 dfs(v);
97 }
98 }
99 }
100 path.pop_back();
101 };
102 dfs(from);
103 return paths;
104 }
