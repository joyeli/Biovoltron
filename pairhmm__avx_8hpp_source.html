38  inline auto getData(const std::vector<SamRecord<>>& readDataArray,

40  const auto numReads = readDataArray.size();
41  const auto numHaplotypes = haplotypeDataArray.size();

44  auto haplotypeLengths = std::vector<int>{};

46  auto total_hap_length = 0;
47  auto total_read_length = 0;
48 
49  // get haplotypes
50  for (auto i = 0; i < numHaplotypes; i++) {
51  const auto length = haplotypeDataArray[i].seq.length();
52  haplotypes.push_back(haplotypeDataArray[i].seq.data());
53  haplotypeLengths.push_back(length);
54  total_hap_length += length;
55  }
56 
57  // get reads and create testcases
58  for (auto r = 0; r < numReads; r++) {
59  const auto length = readDataArray[r].seq.length();

61  const auto readLength = static_cast<int>(length);
62  const char* insGops = readDataArray[r].insertion_gop().data();
63  const char* delGops = readDataArray[r].deletion_gop().data();
64  const char* gapConts = readDataArray[r].overall_gcp().data();
65  const char* readQuals = readDataArray[r].qual.data();
66  total_read_length += length;
67 
68  auto n_testcases = std::vector<testcase>{};
69  for (auto h = 0; h < numHaplotypes; h++)
70  n_testcases.push_back({.rslen = readLength,
71  .haplen = haplotypeLengths[h],
72  .q = readQuals,
73  .i = insGops,

75  .c = gapConts,

139 
140  const auto result_float

142  if (result_float < MIN_ACCEPTED) {

144  result_final
145  = std::log10(result_double) - g_ctxd.LOG10_INITIAL_CONSTANT;
146  } else {
147  result_final = ::log10f(result_float) - g_ctxf.LOG10_INITIAL_CONSTANT;
148  }
149  likelihoodArray[i][j] = result_final;
150  SPDLOG_DEBUG("result = {}", result_final);
151  }

156  SPDLOG_DEBUG("Exit");
