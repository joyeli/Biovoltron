127  // set max for all leaves
128  for (auto idx = 0; idx < tree.size(); idx += 2)
129  tree[idx].max = tree[idx].end;
130 
131  // set max for imaginary node
132  auto last_i = tree.size() - 1;
133  auto last_max = tree[last_i].end;
134 
135  // set max level-by-level bottom-up
136  auto level = 1;
137  for (; (1u << level) <= tree.size(); level++) {
138  const auto first_node = (1u << level) - 1;
139  const auto step = (1u << (level + 1));
140  for (auto idx = first_node; idx < tree.size(); idx += step) {
141  const auto right_max = right_child(idx, level) < tree.size() ?
142  tree[right_child(idx, level)].max :
143  last_max;
144  const auto left_max = tree[left_child(idx, level)].max;
145  tree[idx].max = std::max({right_max, left_max, tree[idx].end});
146  }
147  // update max for imaginary node
148  last_i = parent(last_i, level);
149  if (last_i < tree.size() && tree[last_i].max > last_max)
150  last_max = tree[last_i].max;
151  }

153  return level - 1;
154  }
