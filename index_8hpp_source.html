93 auto get_chr_size(const std::string_view chr) const {
94 auto target_itr = std::ranges::find(chr_bounds, chr, &ChromBound::chrom);
95 if (target_itr == chr_bounds.end())

98 return target_itr->last_elem_pos + 1;

111 auto get_intervals(size_type beg, size_type end, size_type read_len) const {
112 auto intvs = std::vector<Interval>{};
113 intvs.reserve(end - beg);
114
115 for (auto i = beg; i < end; i++) {
116 size_type pos;
117 // check the comment in declaration in the struct.
118 // if constexpr (SA_INTV == 1) pos = sa_[i];
119 // else pos = compute_sa(i);
120 pos = compute_sa(i);
121
122 auto first = std::ranges::lower_bound(
123 chr_bounds, pos, {}, &ChromBound::last_elem_pos);
124 auto last = std::ranges::lower_bound(
125 chr_bounds, pos+read_len-1, {}, &ChromBound::last_elem_pos);
126
127 // Skip pos out of bound or cross chrom mapping.
128 if (first == chr_bounds.end() || first != last)
129 continue;
130
131 pos -= (first == chr_bounds.begin()) ? 0 : (first-1)->last_elem_pos + 1;
132 intvs.emplace_back(first->chrom, pos, pos + read_len);
133 }
134 return intvs;
135 }
