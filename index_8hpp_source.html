100  auto get_chr_size(const std::string_view chr) const {
101  auto target_itr = std::ranges::find(chr_bounds, chr, &ChromBound::chrom);
102  if (target_itr == chr_bounds.end())

104  if (target_itr == chr_bounds.begin())
105  return target_itr->last_elem_pos + 1;

118  auto get_intervals(size_type beg, size_type end, size_type read_len) const {
119  auto intvs = std::vector<Interval>{};
120  intvs.reserve(end - beg);
121 
122  for (auto pos : get_offsets(beg, end)) {
123  auto first = std::ranges::lower_bound(
124  chr_bounds, pos, {}, &ChromBound::last_elem_pos);
125  auto last = std::ranges::lower_bound(
126  chr_bounds, pos+read_len-1, {}, &ChromBound::last_elem_pos);
127 
128  // Skip pos out of bound or cross chrom mapping.
129  if (first == chr_bounds.end() || first != last)
130  continue;
131 
132  pos -= (first == chr_bounds.begin()) ? 0 : (first-1)->last_elem_pos + 1;
133  intvs.emplace_back(first->chrom, pos, pos + read_len);
134  }
135  return intvs;
136  }
