309 void cAdapter::Init(char * seq, size_t sLen, TRIM_MODE trimMode)
310 {
311  int i;
312  // construct sequence
313  this->len = (int(sLen) > MAX_ADAPTER_LEN) ? MAX_ADAPTER_LEN : sLen;
314  skewer::gzstrncpy(sequence, seq, len);
315  this->trimMode = trimMode;
316 
317  // construct mismatch bits
318  int code, code2;
319  uint64_t bits;
320  for(code=0; code<CD_CNT; code++){
321  bits = 0;
322  for(i=int(len)-1; i>=0; i--){
323  code2 = codeMap[uchar(sequence[i])];
324  bits = (bits << 1) | chrVadp[code][code2];
325  }
326  matchBits[code] = ~bits;
327  }
328 }

330 void cAdapter::Init2(char * seq, size_t sLen)
331 {
332  int i;
333  // construct sequence
334  if(int(sLen) > MAX_ADAPTER_LEN){
335  // seq += (sLen - MAX_ADAPTER_LEN);
336  // the head instead of the tail should be used
337  this->len = MAX_ADAPTER_LEN;
338  }
339  else{
340  this->len = sLen;
341  }
342  for(i=0; i<int(len); i++)
343  sequence[i] = character[complement[codeMap[uchar(seq[len-1-i])]]];
344  sequence[len] = '\0';
345  this->trimMode = TRIM_TAIL;
346 
347  // construct mismatch bits
348  int code, code2;
349  uint64_t bits;
350  for(code=CD_BASIC_CNT-1; code>=0; code--){
351  bits = 0;
352  for(i=int(len)-1; i>=0; i--){
353  code2 = codeMap[uchar(sequence[i])];
354  bits = (bits << 1) | chrVadp[code][code2];
355  }
356  matchBits[code] = ~bits;
357  }
358  for(bits=~bits,code=CD_BASIC_CNT; code<CD_CNT; code++){
359  matchBits[code] = bits;
360  }
361 }

363 inline void cAdapter::UPDATE_COLUMN(std::deque<ELEMENT> & queue, uint64_t &d0bits, uint64_t &lbits, uint64_t &unbits, uint64_t &dnbits, double &penal, double &dMaxPenalty, int &iMaxIndel)
364 {
365  int i;
366  double score;
367  uint64_t bits = ~lbits | d0bits;
368  for(bits>>=1,i=1; i<int(queue.size())-1; i++,bits>>=1){
369  if((bits & 0x01) == 0){
370  if(cMatrix::bSensitive){
371  score = queue[i].score + (penal - cMatrix::dDelta);
372  if( (queue[i-1].score < score) && (queue[i-1].nIndel < iMaxIndel) ){
373  if( (queue[i+1].score < score) && (queue[i+1].nIndel < iMaxIndel) ){
374  if(queue[i-1].score < queue[i+1].score){
375  queue[i] = queue[i-1];
376  dnbits |= (1L << (i-1));
377  }
378  else{
379  queue[i] = queue[i+1];
380  unbits |= (1L << (i+1));
381  }
382  }
383  else{
384  queue[i] = queue[i-1];
385  dnbits |= (1L << (i-1));
386  }
387  queue[i].nIndel++;
388  }
389  else{
390  if( (queue[i+1].score < score) && (queue[i+1].nIndel < iMaxIndel) ){
391  queue[i] = queue[i+1];
392  unbits |= (1L << (i+1));
393  queue[i].nIndel++;
394  }
395  else{
396  queue[i].score = score;
397  }
398  }
399  queue[i].score += cMatrix::dDelta;
400  }
401  else{ // !cMatrix::bSensitive
402  queue[i].score += penal;
403  }
404  if(queue[i].score >= dMaxPenalty){
405  lbits &= ~(1L << i);
406  }
407  }
408  }
409  if(queue.size() > 1){
410  if((bits & 0x01) == 0){
411  if(cMatrix::bSensitive){
412  if( (queue[i-1].nIndel < iMaxIndel) && (queue[i-1].score + cMatrix::dDelta < queue[i].score + penal) ){
413  queue[i] = queue[i-1];
414  dnbits |= (1L << (i-1));
415  queue[i].score += cMatrix::dDelta;
416  queue[i].nIndel++;
417  }
418  else{
419  queue[i].score += penal;
420  }
421  }
422  else{ // !cMatrix::bSensitive
423  queue[i].score += penal;
424  }
425  if(queue[i].score >= dMaxPenalty){
426  lbits &= ~(1L << i);
427  }
428  }
429  for(; i>0; i--){
430  if(queue.back().score < dMaxPenalty) break;
431  queue.pop_back();
432  }
433  }

436 bool cAdapter::align(char * read, size_t rLen, uchar * qual, size_t qLen, cElementSet &result, int bc, bool bBestAlign)
437 {
438  bool bDetermined = false;
439  ELEMENT elem;
440  double dMaxPenalty = cMatrix::dPenaltyPerErr * len + 0.001;
441  int iMaxIndel = ceil(cMatrix::dEpsilonIndel * len);
442  int minK = bBestAlign ? ((cMatrix::iMinOverlap >= (int)(len - iMaxIndel + 1)) ? (int)(len - iMaxIndel + 1) : cMatrix::iMinOverlap) : 1;
443  double dMu = (bc >= 0) ? cMatrix::dMu : MIN_PENALTY;
444 
445  std::deque<ELEMENT> queue;
446  ELEMENT element;
447  double score;
448  uint64_t legalBits = 0;
449  int i, j, jj;
450  element.idx.bc = bc + 1;
451  if(trimMode & TRIM_HEAD){
452  for(i=1; i<=int(len)-minK; i++){
453  element.idx.pos = -i;
454  element.score = cMatrix::dPenaltyPerErr * i;
455  element.nIndel = 0;
456  queue.push_back(element);
457  legalBits = (legalBits << 1) | 1;
458  }
459  }
460  else{
461  for(i=1,score=cMatrix::dDelta; i<int(len); i++,score+=cMatrix::dDelta){
462  if(i > iMaxIndel) break;
463  element.idx.pos = -i;
464  element.score = score;
465  element.nIndel = i;
466  queue.push_back(element);
467  legalBits = (legalBits << 1) | 1;
468  }
469  }
470  element.nIndel = 0;
471  uint64_t mbits, xbits, unbits, dnbits, d0bits;
472  unbits = dnbits = 0L;
473  double penal;
474  for(j=0; j<int(rLen); j++){
475  jj = j;
476  mbits = matchBits[codeMap[uchar(read[jj])]];
477  penal = ((qLen > 0) ? cMatrix::penalty[qual[jj]] : dMu);
478 
479  element.idx.pos = j;
480  element.score = ((mbits & 0x01) == 0) ? penal : 0;
481  queue.push_front(element);
482 
483  xbits = mbits | unbits;
484  dnbits <<= 1;
485  unbits <<= 1;
486  d0bits = ((dnbits + (xbits & dnbits)) ^ dnbits) | xbits;
487  legalBits = (legalBits << 1) | 1;
488 
489  UPDATE_COLUMN(queue, d0bits, legalBits, unbits, dnbits, penal, dMaxPenalty, iMaxIndel);
490 
491  dnbits &= d0bits;
492  unbits &= d0bits;
493 
494  if(queue.size() == len){
495  if(bBestAlign){
496  if(trimMode == TRIM_HEAD){
497  i = (queue.back().idx.pos < 0) ? (len + queue.back().idx.pos) : len;
498  if( !bDetermined || (i * cMatrix::dMu - queue.back().score) > elem.score * (i+1) ){
499  elem = queue.back();
500  dMaxPenalty = elem.score;
501  elem.score = (i * cMatrix::dMu - elem.score) / (i+1); // normalization
502  elem.idx.pos = rLen - 1 - j;
503  }
504  }
505  else{
506  elem = queue.back();
507  dMaxPenalty = elem.score + ((trimMode == TRIM_TAIL) ? EPSILON : 0);
508  elem.score = (len * cMatrix::dMu - elem.score) / (len + 1); // normalization
509  }
510  bDetermined = true;
511  if(dMaxPenalty == 0) break;
512  }
513  else{
514  elem = queue.back();
515  elem.score = len * cMatrix::dMu - elem.score; // normalization
516  result.insert(elem);
517  }
518 
519  if (cMatrix::bIsLowComplexity)
520  break;
521 
522  queue.pop_back();
523  }
524  }
525  if(dMaxPenalty > 0){ // not the case of "perfect match for single-end reads trimming"
526  if(bBestAlign){
527  if(trimMode & TRIM_TAIL){
528  dMaxPenalty = (cMatrix::dPenaltyPerErr * queue.size() + 0.001);
529  for(i=queue.size(); i>=minK; i--, dMaxPenalty-=cMatrix::dPenaltyPerErr){
530  if(dMaxPenalty <= 0) break;
531  if(queue.back().score < dMaxPenalty){
532  if(!bDetermined || ((i * cMatrix::dMu - queue.back().score) > elem.score * (i+1)) ){
533  elem = queue.back();
534  dMaxPenalty = elem.score;
535  elem.score = (i * cMatrix::dMu - elem.score) / (i+1); // normalization
536  bDetermined = true;
537  }
538  }
539  queue.pop_back();
540  }
541  }
542  else{
543  dMaxPenalty -= (len - queue.size()) * cMatrix::dDelta;
544  iMaxIndel -= (len - queue.size());
545  for(i=queue.size(); i>=minK; i--, dMaxPenalty-=cMatrix::dDelta, iMaxIndel--){
546  if( (dMaxPenalty <= 0) || (iMaxIndel < 0) ) break;
547  if( (queue.back().score < dMaxPenalty) && (queue.back().nIndel <= iMaxIndel) ){
548  if(!bDetermined || ((i * cMatrix::dMu - queue.back().score) > elem.score * (i+1)) ){
549  elem = queue.back();
550  dMaxPenalty = elem.score;
551  elem.score = (i * cMatrix::dMu - elem.score) / (i+1); // normalization
552  elem.idx.pos = -(len - i);
553  bDetermined = true;
554  }
555  }
556  queue.pop_back();
557  }
558  }
559  }
560  else{
561  dMaxPenalty = cMatrix::dPenaltyPerErr * queue.size() + 0.001;
562  for(i=queue.size(); i>=minK; i--, dMaxPenalty-=cMatrix::dPenaltyPerErr){
563  if(queue.back().score < dMaxPenalty){
564  elem = queue.back();
565  elem.score = i * cMatrix::dMu - elem.score; // normalization
566  result.insert(elem);
567  }
568  queue.pop_back();
569  }
570  }
571  }
572 
573  if(bDetermined){
574  result.clear();
575  result.insert(elem);
576  }
577 
578  return bDetermined;

632 bool cMatrix::CalcRevCompScore(char * seq, char * seq2, int len, uchar * qual, uchar * qual2, size_t qLen, double &score)
633 {
634  double dMaxPenalty = dPenaltyPerErr * len;
635  double penal;
636  CODE code, code2;
637  if(len <= 0){
638  score = (qLen > 0) ? (dMu * qLen / 2) : 0.0;
639  // prefer to detect empty reads even if an error ratio of 0.5 is specified
640  return true;
641  }
642  score = 0.0;
643  for(int i=0; i<len; i++){
644  code = codeMap[uchar(seq[i])];
645  code2 = complement[codeMap[uchar(seq2[len-1-i])]];
646  penal = scoring[code][code2];
647  if(penal > 0.0){
648  if(qLen > 0){
649  if(cMatrix::penalty[qual[i]] <= cMatrix::penalty[qual2[len-1-i]]){
650  penal *= cMatrix::penalty[qual[i]];
651  }
652  else{
653  penal *= cMatrix::penalty[qual2[len-1-i]];
654  }
655  }
656  else{
657  penal *= dMu;
658  }
659  score += penal;
660  if(score > dMaxPenalty){
661  return false;
662  }
663  }
664  }
665  score = len * dMu - score; // normalization
666  return true;
