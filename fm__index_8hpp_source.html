171  auto
172  compute_occ(char_type c, size_type i) const {
173  const auto occ1_beg = i / OCC1_INTV;
174  const auto occ2_beg = i / OCC2_INTV;
175  auto beg = occ2_beg * OCC2_INTV;
176  auto cnt = size_type{};
177  const auto pass_pri = c == 0 && beg <= pri_ && pri_ < i;
178  const auto run = (i - beg) / 4;
179  for (auto j = size_type{}; j < run; j++) {
180  cnt += cnt_table[bwt_.data()[beg / 4]][c];
181  beg += 4;
182  }
183  for (; beg < i; beg++)
184  if (bwt_[beg] == c)
185  cnt++;
186  return occ_.first[occ1_beg][c] + occ_.second[occ2_beg][c] + cnt - pass_pri;
187  }

194  auto compute_b_occ(size_type i) const {
195  if constexpr (SA_INTV == 1)
196  return i;
197  else {
198  const auto b_occ_beg = i / B_OCC_INTV;
199  auto beg = b_occ_beg * B_OCC_INTV;
200  auto cnt = size_type{};
201 
202  const auto run = (i - beg) / 64;
203  for (auto j = size_type{}; j < run; j++) {
204  cnt += std::popcount(b_.data()[beg / 64]);
205  beg += 64;
206  }
207 
208  auto mask = (1ull << i - beg) - 1;
209  cnt += std::popcount(b_.data()[beg / 64] & mask);
210  return b_occ_[b_occ_beg] + cnt;

241  auto
242  build_lookup() {
243  auto lookup_size = size_type{1} << LOOKUP_LEN * 2;
244  lookup_.resize(lookup_size);
245  lookup_.push_back(bwt_.size());
246 
247  const auto set_value = [this](const auto beg_i, const auto end_i) {
248  const auto remain = (end_i - beg_i) % 2;
249  const auto last_i = end_i - remain;
250  for (auto i = beg_i; i < last_i; i += 2) {
251  const auto seed = Codec::rhash(i, LOOKUP_LEN);
252  const auto [beg, end, offset]
253  = compute_range(seed, 0, lookup_.back(), 0);
254  lookup_[i] = beg;
255  lookup_[i + 1] = end;
256  }
257  if (remain) {
258  const auto seed = Codec::rhash(last_i, LOOKUP_LEN);
259  const auto [beg, end, offset]
260  = compute_range(seed, 0, lookup_.back(), 0);
261  lookup_[last_i] = beg;
262  }
263  };
264 
265  const auto thread_n = std::thread::hardware_concurrency();
266  const auto batch_size = lookup_size / thread_n;
267  auto workers = std::vector<std::thread>{};
268  for (auto i = 0; i < thread_n; i++) {
269  workers.emplace_back(set_value, i * batch_size, (i + 1) * batch_size);
270  }
271  set_value(thread_n * batch_size, lookup_size);
272  for (auto& worker : workers) worker.join();
273  }

290  void
291  build_occ(istring_view ref, const auto &ori_sa) {
292  auto& [occ1, occ2] = occ_;
293  occ1.resize(ori_sa.size() / OCC1_INTV + 1);
294  occ2.resize(ori_sa.size() / OCC2_INTV + 1);
295 #pragma omp parallel for
296  for (auto beg = size_type{}; beg < ori_sa.size(); beg += OCC1_INTV) {
297  auto &cnt = occ1[beg / OCC1_INTV];
298  for (auto i = beg; i < beg + OCC1_INTV; i++) {
299  if (i % OCC2_INTV == 0)
300  for (int j = 0; j < 4; j++)
301  occ2[i / OCC2_INTV][j] = cnt[j];
302  if (i == ori_sa.size())
303  break;
304  const auto sa_v = ori_sa[i];
305  if (sa_v != 0)
306  cnt[ref[sa_v - 1]]++;
307  }
308  }
309 
310  for (auto &x : occ1) {
311  for (int j = 0; j < 4; j++) {
312  cnt_[j] += x[j];
313  x[j] = cnt_[j] - x[j];
314  }
315  }
316 
317  auto sum = size_type{1};
318  for (auto& x : cnt_) {
319  sum += x;
320  x = sum - x;
321  }
322  }

345  void build_sa(istring_view ref, const auto &ori_sa) {
346  if constexpr (SA_INTV == 1) {
347  sa_ = ori_sa;
348  return ;
349  }

354  for (auto beg = size_type{}; beg < ori_sa.size(); beg += B_OCC_INTV) {
355  auto &cnt = b_occ_[beg / B_OCC_INTV];
356  for (auto i = beg; i < beg + B_OCC_INTV; i++) {
357  if (i == ori_sa.size())
358  break;
359  const auto sa_v = ori_sa[i];
360  b_[i] = sa_v % SA_INTV == 0;
361  cnt += b_[i];
362  }
363  }

365  auto sum = size_type{};
366  for (auto &x : b_occ_) {
367  sum += x;
368  x = sum - x;
369  }

372 #pragma omp parallel for
373  for (auto beg = size_type{}; beg < ori_sa.size(); beg += B_OCC_INTV) {
374  auto ptr = b_occ_[beg / B_OCC_INTV];
375  for (auto i = beg; i < beg + B_OCC_INTV; i++) {
376  if (i == ori_sa.size())
377  break;
378  const auto sa_v = ori_sa[i];
379  if (sa_v % SA_INTV == 0) {
380  sa_[ptr++] = sa_v;
381  }
382  }
383  }

469  auto
470  get_offsets(size_type beg, size_type end) const {
471  if constexpr (SA_INTV == 1)
472  return std::span{&sa_[beg], end - beg};
473  else {
474  auto offsets = std::vector<size_type>{};
475  offsets.reserve(end - beg);
476 
477  constexpr auto MAX_DEPTH = SA_INTV;
478 
479  auto q = std::queue<std::tuple<size_type, size_type, int>>{};
480  q.emplace(beg, end, 0);
481 
482  auto enqueue = [&q](auto beg, auto end, auto dep) {
483  if (beg != end)
484  q.emplace(beg, end, dep);
485  };
486 
487  while (q.size() and offsets.size() < end - beg) {
488  auto [cur_beg, cur_end, cur_dep] = q.front(); q.pop();
489 
490  // add offset from sampled sa value
491  const auto b_occ_cur_beg = compute_b_occ(cur_beg);
492  const auto b_occ_cur_end = compute_b_occ(cur_end);
493  for (auto i = b_occ_cur_beg; i < b_occ_cur_end; i++) {
494  offsets.push_back(sa_[i] + cur_dep);
495  }
496 
497  const auto nxt_dep = cur_dep + 1;
498  if (nxt_dep == MAX_DEPTH)
499  continue;
500 
501  if (cur_beg + 1 == cur_end) {
502  const auto nxt_beg = lf(bwt_[cur_beg], cur_beg);
503  const auto nxt_end = nxt_beg + 1;
504  q.emplace(nxt_beg, nxt_end, nxt_dep);
505  } else {
506  [&]<auto ... Idx>(std::index_sequence<Idx...>) {
507  auto bg = std::array{lf(Idx, cur_beg)...};
508  auto ed = std::array{lf(Idx, cur_end)...};
509  (enqueue(bg[Idx], ed[Idx], nxt_dep), ...);
510  } (std::make_index_sequence<4>{});
511  }
512  }

514  }
