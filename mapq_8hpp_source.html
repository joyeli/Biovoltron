59 static auto
60 mem_approx_mapq_se(MemAln aln) {
61  // Decompose alignment metrics
62  const auto [score, csub, sub_score, l, sub_n, frac_rep] = aln;
63 
64  // Use sub_score if available, otherwise default to 20
65  auto sub = sub_score ? sub_score : 20;
66  sub = csub > sub ? csub : sub;
67  // If suboptimal score is not less than best score, mapping is unreliable
68  if (sub >= score)
69  return 0;
70 
71  // Calculate identity and scaling factor
72  const auto identity = 1. - (l - score) / 5. / l;
73  // Scale by read length; short reads use 1, longer reads use log scaling
74  auto tmp = l < 50 ? 1. : 3.912 / std::log(l);
75  tmp *= identity * identity;
76 
77  // Main MAPQ formula: higher score difference and identity yield higher MAPQ
78  auto mapq = static_cast<int>(6.02 * (score - sub) * tmp * tmp + .499);
79  // Penalize for multiple suboptimal alignments
80  if (sub_n > 0)
81  mapq -= static_cast<int>(4.343 * std::log(sub_n + 1) + .499);
82 
83  // Clamp MAPQ to [0, 60]
84  if (mapq > 60) mapq = 60;
85  if (mapq < 0) mapq = 0;
86 
87  // Penalize for repetitive content
88  mapq = static_cast<int>(mapq * (1. - frac_rep) + .499);
89  return mapq;
90 }

117 mem_mapq_pe(MemAln p0, MemAln p1, int score_un, int o, int subo, int n_sub) {
118  // Paired-end MAPQ calculation: combine score difference, repetitive content, and suboptimal hits
119  subo = subo > score_un ? subo : score_un;
120  auto q_pe = raw_mapq(o - subo);
121  if (n_sub > 0)
122  q_pe -= static_cast<int>(4.343 * std::log(n_sub + 1) + .499);
123  if (q_pe < 0) q_pe = 0;
124  if (q_pe > 60) q_pe = 60;
125 
126  // Penalize for repetitive content in both mates
127  q_pe = static_cast<int>(q_pe * (1. - .5 * (p0.frac_rep + p1.frac_rep)) + .499);
128 
129  // Compute single-end MAPQ for each mate
130  auto q_se0 = mem_approx_mapq_se(p0);
131  auto q_se1 = mem_approx_mapq_se(p1);
132 
133  // Adjust single-end MAPQ if paired-end MAPQ is higher
134  q_se0 = q_se0 > q_pe ? q_se0 : q_pe < q_se0 + 40 ? q_pe : q_se0 + 40;
135  q_se1 = q_se1 > q_pe ? q_se1 : q_pe < q_se1 + 40 ? q_pe : q_se1 + 40;
136 
137  // Clamp single-end MAPQ to raw MAPQ based on score difference
138  q_se0 = q_se0 < raw_mapq(p0.score - p0.score2) ? q_se0 : raw_mapq(p0.score - p0.score2);
139  q_se1 = q_se1 < raw_mapq(p1.score - p1.score2) ? q_se1 : raw_mapq(p1.score - p1.score2);
140 
141  if (q_se0 > 60) q_se0 = 60;
142  if (q_se1 > 60) q_se1 = 60;
143 
144  return std::pair{q_se0, q_se1};
